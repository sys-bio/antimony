model thomsen_neubig_1989_version01__environment(time_)
end

model thomsen_neubig_1989_version01__R(R, D, DR, k1, k1_, time_)

  // Rate Rules:
  R' = -((k1 * D * R)) + k1_ * DR;
  D' = -((k1 * D * R)) + k1_ * DR;

  // Variable initializations:
  R = 0.001;
  D = 0.01;
end

model thomsen_neubig_1989_version01__DR(k1, k1_, k2, k2_, k5, D, R, DR, DRG_GDP, DRG_GTP, G_GDP, time_)

  // Rate Rules:
  DR' = k1 * D * R - k1_ * DR - k2 * DR * G_GDP + k2_ * DRG_GDP + k5 * DRG_GTP;
end

model thomsen_neubig_1989_version01__DRG_GDP(k2, k2_, k3, k3_, DR, DRG, DRG_GDP, G_GDP, GDP, time_)

  // Rate Rules:
  DRG_GDP' = k2 * DR * G_GDP + k3_ * DRG * GDP - (k2_ + k3) * DRG_GDP;
end

model thomsen_neubig_1989_version01__DRG(k3, k3_, k4, k4_, DRG, DRG_GDP, DRG_GTP, GDP, GTP, time_)

  // Rate Rules:
  DRG' = k3 * DRG_GDP + k4_ * DRG_GTP - (k3_ * GTP + k4 * GTP * DRG);
end

model thomsen_neubig_1989_version01__DRG_GTP(k4, k4_, k5, DRG_GTP, DRG, GTP, time_)

  // Rate Rules:
  DRG_GTP' = k4 * DRG * GTP - (k4_ + k5) * DRG_GTP;
end

model thomsen_neubig_1989_version01__Gstar_GTP(k5, k6, DRG_GTP, Gstar_GTP, time_)

  // Rate Rules:
  Gstar_GTP' = k5 * DRG_GTP - k6 * Gstar_GTP;
end

model thomsen_neubig_1989_version01__G_GDP(G_GDP, Gstar_GTP, k6, k2, time_, DRG_GTP)

  // Rate Rules:
  G_GDP' = k6 * Gstar_GTP - k2 * G_GDP;
end

model thomsen_neubig_1989_version01__rate_variables(k1, k1_, k2, k2_, k3, k3_, k4, k4_, k5, k6)

  // Variable initializations:
  k1 = 5000000;
  k1_ = 0.5;
  k2 = 0.1;
  k2_ = 0.1;
  k3 = 0.1;
  k3_ = 0.0001;
  k4 = 100000000;
  k4_ = 0.1;
  k5 = 0.05;
  k6 = 0.1;
end

model thomsen_neubig_1989_version01__GXP(GDP, GTP)

  // Variable initializations:
  GDP = 1000;
  GTP = 10;
end

model thomsen_neubig_1989_version01____main()

  // Sub-modules, and any changes to those submodules:
  environment: thomsen_neubig_1989_version01__environment(time_);
  R: thomsen_neubig_1989_version01__R(R0, D, DR0, k1, k1_, time_);
  DR: thomsen_neubig_1989_version01__DR(k1, k1_, k2, k2_, k5, D, R0, DR0, DRG_GDP0, DRG_GTP0, G_GDP0, time_);
  DRG_GDP: thomsen_neubig_1989_version01__DRG_GDP(k2, k2_, k3, k3_, DR0, DRG0, DRG_GDP0, G_GDP0, GDP, time_);
  DRG: thomsen_neubig_1989_version01__DRG(k3, k3_, k4, k4_, DRG0, DRG_GDP0, DRG_GTP0, GDP, GTP, time_);
  DRG_GTP: thomsen_neubig_1989_version01__DRG_GTP(k4, k4_, k5, DRG_GTP0, DRG0, GTP, time_);
  Gstar_GTP: thomsen_neubig_1989_version01__Gstar_GTP(k5, k6, DRG_GTP0, Gstar_GTP0, time_);
  G_GDP: thomsen_neubig_1989_version01__G_GDP(G_GDP0, Gstar_GTP0, k6, k2, time_, DRG_GTP0);
  rate_variables: thomsen_neubig_1989_version01__rate_variables(k1, k1_, k2, k2_, k3, k3_, k4, k4_, k5, k6);
  GXP: thomsen_neubig_1989_version01__GXP(GDP, GTP);
end

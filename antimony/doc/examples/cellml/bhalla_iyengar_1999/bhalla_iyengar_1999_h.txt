model bi_model_1999_version01__PIPtoDAG(t, CaPLCg, CaPLCg_star)

  // Rate Rules:
  PIP2' = -(CaPLCg) * Vmax2 * PIP2 / (km2 + PIP2) - CaPLCg_star * Vmax6 * PIP2 / (km6 + PIP2);
  DAG' = CaPLCg * Vmax2 * PIP2 / (km2 + PIP2) + CaPLCg_star * Vmax6 * PIP2 / (km6 + PIP2);
  IP3' = CaPLCg * Vmax2 * PIP2 / (km2 + PIP2) + CaPLCg_star * Vmax6 * PIP2 / (km6 + PIP2);

  // Variable initializations:
  km2 = 97;
  km6 = 19.79166667;
  Vmax2 = 14;
  Vmax6 = 57;
  PIP2 = 10;
  DAG = 0;
  IP3 = 0;
end

model bi_model_1999_version01__PLC(t, CaPLCg, CaPLCg_star, EGF_EGFR)

  // Rate Rules:
  PLCg' = -(Ca) * PLCg * kf1 + CaPLCg * kb1;
  PLCg_star' = -(Ca) * PLCg_star * kf5 + kb5 * CaPLCg_star;
  CaPLCg' = -((-(Ca) * PLCg * kf1 + CaPLCg * kb1)) + kf3 * CaPLCg_star - Vmax4 * EGF_EGFR * CaPLCg / (km4 + CaPLCg);
  CaPLCg_star' = -((kf3 * CaPLCg_star - Vmax4 * EGF_EGFR * CaPLCg / (km4 + CaPLCg))) + Ca * PLCg_star * kf5 - kb5 * CaPLCg_star;
  Ca' = -(Ca) * PLCg * kf1 + CaPLCg * kb1 + kb5 * CaPLCg_star - Ca * PLCg_star * kf5;

  // Variable initializations:
  kf1 = 0.0003;
  kb1 = 10;
  kf3 = 0.05;
  kf5 = 0.00002;
  kb5 = 10;
  km4 = 0.33333333333;
  Vmax4 = 0.2;
  PLCg = 0.82;
  PLCg_star = 0;
  CaPLCg = 0;
  CaPLCg_star = 0;
  Ca = 0.1;
end

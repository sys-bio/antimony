end

model bi_model_1999_version01__PIPtoDAG(t, CaPLC, CaGqPLC)

  // Rate Rules:
  PIP2' = -(CaPLC) * Vmax7 * PIP2 / (km7 + PIP2) - CaGqPLC * Vmax6 * PIP2 / (km6 + PIP2);
  DAG' = CaPLC * Vmax7 * PIP2 / (km7 + PIP2) - kf8 * DAG;
  IP3' = CaGqPLC * Vmax6 * PIP2 / (km6 + PIP2) - kf9 * IP3;
  Inositol' = kf9 * IP3;
  PC' = kf8 * DAG;

  // Variable initializations:
  kf8 = 0.15;
  kf9 = 2.5;
  km6 = 5;
  Vmax6 = 48;
  km7 = 19.84166667;
  Vmax7 = 10;
  PIP2 = 10;
  DAG = 0;
  IP3 = 0;
  Inositol = 0;
  PC = 0;
end

model bi_model_1999_version01__PLC(t, CaGqPLC, CaPLC)

  // Rate Rules:
  PLC' = -(PLC) * Ca * kf1 + CaPLC * kb1 - PLC * Gq * kf2 + GqPLC * kb2;
  Gq' = -(PLC) * Gq * kf2 + GqPLC * kb2 + CaGqPLC * kb4 - CaPLC * Gq * kf4;
  Ca' = -(PLC) * Ca * kf1 + CaPLC * kb1 - Ca * GqPLC * kf3 + CaGqPLC * kb3;
  GqPLC' = PLC * Gq * kf2 - GqPLC * kb2 - Ca * GqPLC * kf3 + CaGqPLC * kb3;
  CaGqPLC' = GqPLC * Ca * kf3 - CaGqPLC * kb3 + CaPLC * Gq * kf4 - CaGqPLC * kb4 - CaGqPLC * kf5;
  CaPLC' = PLC * Ca * kf1 - CaPLC * kb1 + CaGqPLC * kf5 + CaGqPLC * kb4 - CaPLC * Gq * kf4;

  // Variable initializations:
  kf1 = 0.000005;
  kb1 = 1;
  kf2 = 0.0000042;
  kb2 = 1;
  kf3 = 0.00005;
  kb3 = 1;
  kf4 = 0.0000042;
  kb4 = 1;
  kf5 = 0.0133;
  PLC = 0.8;
  Gq = 0.15;
  Ca = 0.1;
  GqPLC = 0;
  CaGqPLC = 0;
  CaPLC = 0;
end

model bi_model_1999_version01____main()

  // Sub-modules, and any changes to those submodules:
  environment: bi_model_1999_version01__environment(t);
  PIPtoDAG: bi_model_1999_version01__PIPtoDAG(t, CaPLC, CaGqPLC);
  PLC: bi_model_1999_version01__PLC(t, CaGqPLC, CaPLC);
end

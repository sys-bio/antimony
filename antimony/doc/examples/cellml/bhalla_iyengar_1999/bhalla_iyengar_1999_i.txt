model bi_model_1999_version01__GEF(t, GEF_inact, PKA, PKC, CaM_GEF, Gby_GEF)

  // Rate Rules:
  GEF_inact' = PKA * Vmax1 * GEF / (km1 + GEF) + PKC * Vmax6 * GEF / (km6 + GEF) - kf2 * GEF_inact - kf7 * GEF_inact;
  GEF' = -((PKA * Vmax1 * GEF / (km1 + GEF) + PKC * Vmax6 * GEF / (km6 + GEF) - kf2 * GEF_inact - kf7 * GEF_inact)) - kf5 * Ca4CAM * GEF + kb5 * CaM_GEF - kf8 * GEF * Gby + kb8 * Gby_GEF;
  Ca4CAM' = -((-(kb5) * CaM_GEF + kf5 * GEF * Ca4CAM));
  Gby' = -(kf8) * GEF * Gby + kb8 * Gby_GEF;
  CaM_GEF' = -(kb5) * CaM_GEF + kf5 * GEF * Ca4CAM;
  Gby_GEF' = -((-(kf8) * GEF * Gby + kb8 * Gby_GEF));

  // Variable initializations:
  kf2 = 1;
  kb2 = 0;
  kf5 = 0.0001;
  kb5 = 1;
  kf7 = 1;
  kb7 = 0;
  kf8 = 0.00001;
  kb8 = 1;
  km1 = 7.5;
  Vmax1 = 9;
  km6 = 3.33333333333;
  Vmax6 = 4;
  GEF_inact = 0.1;
  GEF = 0;
  Ca4CAM = 1;
  Gby = 1;
  CaM_GEF = 0;
  Gby_GEF = 0;
end

model bi_model_1999_version01__CaIP3(t)

  // Rate Rules:
  Ca' = -((Ca_pump * Vmax1 * Ca / (km1 + Ca) - Extracell_Leak * Ca_ext * p2 - Cap_channel_star * p4 * Ca_ext)) + Stores_Leak * Ca_stores * p3 + IP3R_star * Ca_stores * p4 - Ca * Ca * Ca_transp * kf7 + Ca2Ca_transp * kb7;
  Ca2Ca_transp' = Ca * Ca * Ca_transp * kf7 - Ca2Ca_transp * kb7 - Ca2Ca_transp * kf6;
  Ca_stores' = -(Stores_Leak) * Ca_stores * p3 - IP3R_star * Ca_stores * p4 - Ca_stores * Cap_channel_star * kf8 + Cap_channel * kb8 + Ca2Ca_transp * kf6;
  Ca_transp' = -(Ca) * Ca * Ca_transp * kf7 + Ca2Ca_transp * kb7 + Ca2Ca_transp * kf6;
  IP3R' = IP3R_star * kb9 - IP3 * IP3 * IP3 * IP3R * kf9;
  IP3R_star' = -(IP3R_star) * kb9 + IP3 * IP3 * IP3 * IP3R * kf9 - IP3R_star * Ca_stores * p5;
  IP3' = IP3R_star * kb9 - IP3 * IP3 * IP3 * IP3R * kf9;
  Ca_ext' = Ca_pump * Vmax1 * Ca / (km1 + Ca) - Extracell_Leak * Ca_ext * p2 - Cap_channel_star * p4 * Ca_ext;
  Cap_channel' = Ca_stores * Cap_channel_star * kf8 - Cap_channel * kb8;
  Cap_channel_star' = -(Ca_stores) * Cap_channel_star * kf8 + Cap_channel * kb8 - Cap_channel_star * p4 * Ca_ext;

  // Variable initializations:
  km1 = 0.2;
  Vmax1 = 72;
  kf6 = 25;
  kf7 = 0.00000001;
  kb7 = 144;
  kf8 = 0.000000000012;
  kb8 = 1;
  kf9 = 0.00000000000000000001;
  kb9 = 1;
  p2 = 0.004;
  p3 = 8;
  p4 = 0.005;
  p5 = 19.2;
  Ca = 0.08;
  Ca2Ca_transp = 0.08;
  Ca_stores = 6.3328;
  Stores_Leak = 0.001;
  Ca_transp = 0.0083838;
  IP3R = 0.016643;
  Ca_pump = 0.00083333;
  Ca_ext = 4000;
  Extracell_Leak = 0.00083333;
  Cap_channel = 0.00083333;
end

end

  Fluo' = k3_ * PrFluo + k1_ * CaFluo - ( k3 * Pr * Fluo + k1 * Ca * Fluo );

  // Variable initializations:

end

model baylor_model_f_2002__Pr(Pr, k3, k3_, k4, k4_, Fluo, PrFluo, CaFluo, CaPrFluo, time_)

  // Rate Rules:
  Pr' = k4_ * CaPrFluo + k3_ * PrFluo - ( k4 * Pr * CaFluo + k3 * Pr * Fluo );

  // Variable initializations:
  Pr = 3000.0;
end

end

model baylor_model_f_2002__CaFluo(CaFluo, k1, k1_, k4, k4_, Fluo, Pr, Ca, CaPrFluo, time_)

  // Rate Rules:
  CaFluo' = k1 * Ca * Fluo + k4_ * CaPrFluo - ( k1_ * CaFluo + k4 * Pr * CaFluo );
end

model baylor_model_f_2002__CaPrFluo(CaPrFluo, k2, k2_, k4, k4_, Ca, CaFluo, Pr, PrFluo, time_)

  // Rate Rules:

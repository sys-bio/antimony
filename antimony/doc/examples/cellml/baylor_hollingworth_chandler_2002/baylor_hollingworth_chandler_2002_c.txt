model baylor_2002c__Ca(Ca, k1, k1_, Parv, CaParv, time_)

  // Rate Rules:
  Ca' = k1_ * CaParv - k1 * Ca * Parv;

  // Variable initializations:

end

model baylor_2002c__Parv(Parv, k1, k1_, k2, k2_, Ca, CaParv, Mg, MgParv, time_)

  // Rate Rules:
  Parv' = k1_ * CaParv + k2_ * MgParv - ( k1 * Ca * Parv + k2 * Mg * Parv );

  // Variable initializations:
  Parv = 1500.0;
end

model baylor_2002c__CaParv(CaParv, k1, k1_, Parv, Ca, time_)

  // Rate Rules:
  CaParv' = k1 * Ca * Parv - k1_ * CaParv;

  // Variable initializations:
  CaParv = 0.01;
end

model baylor_2002c__Mg(Mg, k2, k2_, Parv, MgParv, time_)

  // Rate Rules:
  Mg' = k2_ * MgParv - k2 * Mg * Parv;

  // Variable initializations:
  Mg = 1000.0;
end

model baylor_2002c__MgParv(MgParv, k2, k2_, Parv, Mg, time_)

  // Rate Rules:
  MgParv' = k2 * Mg * Parv - k2_ * MgParv;

  // Variable initializations:
  MgParv = 0.01;
end

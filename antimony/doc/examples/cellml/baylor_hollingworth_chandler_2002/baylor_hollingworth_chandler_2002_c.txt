model baylor_model_f_2002__environment(time_)
end

model baylor_model_f_2002__Ca(Ca, k1, k1_, Parv, CaParv, time_)

  // Rate Rules:
  Ca' = k1_ * CaParv - k1 * Ca * Parv;

  // Variable initializations:
  Ca = 0.05;
end

model baylor_model_f_2002__Parv(Parv, k1, k1_, k2, k2_, Ca, CaParv, Mg, MgParv, time_)

  // Rate Rules:
  Parv' = k1_ * CaParv + k2_ * MgParv - ( k1 + Ca + Parv ) * ( k2 + Mg + Parv );

  // Variable initializations:
  Parv = 1500.0;
end

model baylor_model_f_2002__CaParv(CaParv, k1, k1_, Parv, Ca, time_)

  // Rate Rules:
  CaParv' = k1 * Ca * Parv - k1_ * CaParv;
end

model baylor_model_f_2002__Mg(Mg, k2, k2_, Parv, MgParv, time_)

  // Rate Rules:
  Mg' = k2_ * MgParv - k2 * Mg * Parv;

  // Variable initializations:
  Mg = 1000.0;
end

model baylor_model_f_2002__MgParv(MgParv, k2, k2_, Parv, Mg, time_)

  // Rate Rules:
  MgParv' = k2 * Mg * Parv - k2_ * MgParv;
end

model baylor_model_f_2002__reaction_constants(k1, k1_, k2, k2_)

  // Variable initializations:
  k1 = 4.79E7;
  k1_ = 0.574;
  k2 = 39700.0;
  k2_ = 3.45;
end

model baylor_model_f_2002____main()

  // Sub-modules, and any changes to those submodules:
  environment: baylor_model_f_2002__environment(time_);
  Ca: baylor_model_f_2002__Ca(Ca0, k1, k1_, Parv0, CaParv0, time_);
  Parv: baylor_model_f_2002__Parv(Parv0, k1, k1_, k2, k2_, Ca0, CaParv0, Mg0, MgParv0, time_);
  CaParv: baylor_model_f_2002__CaParv(CaParv0, k1, k1_, Parv0, Ca0, time_);
  Mg: baylor_model_f_2002__Mg(Mg0, k2, k2_, Parv0, MgParv0, time_);
  MgParv: baylor_model_f_2002__MgParv(MgParv0, k2, k2_, Parv0, Mg0, time_);
  reaction_constants: baylor_model_f_2002__reaction_constants(k1, k1_, k2, k2_);
end

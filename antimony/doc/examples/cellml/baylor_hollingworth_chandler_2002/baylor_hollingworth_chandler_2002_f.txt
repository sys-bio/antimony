model baylor_2002f__Ca(Ca, k1, k1_, k2, k2_, Trop, CaTrop, Ca2Trop, time_)

  // Rate Rules:
  Ca' = k1_ * CaTrop + k2_ * Ca2Trop - ( k1 * Ca * Trop + k2 * Ca * CaTrop );

  // Variable initializations:
  Ca = 0.05;
end

model baylor_2002f__Trop(Trop, k1, k1_, Ca, CaTrop, time_)

  // Rate Rules:
  Trop' = k1_ * CaTrop - k1 * Ca * Trop;

  // Variable initializations:
  Trop = 360.0;
end

model baylor_2002f__CaTrop(CaTrop, k1, k1_, k2, k2_, Trop, Ca, Ca2Trop, time_)

  // Rate Rules:
  CaTrop' = k1 * Ca * Trop + k2_ * Ca2Trop - ( k1_ * CaTrop + k2 * Ca * CaTrop );

  // Variable initializations:
  CaTrop = 0.01;
end

model baylor_2002f__Ca2Trop(Ca2Trop, k2, k2_, Ca, CaTrop, time_)

  // Rate Rules:
  Ca2Trop' = k2 * Ca * CaTrop - k2_ * Ca2Trop;

  // Variable initializations:
  Ca2Trop = 0.01;
end

model baylor_2002f__reaction_constants(k1, k1_, k2, k2_)

  // Variable initializations:
  k1 = 2.033E14;
  k1_ = 2642.0;
  k2 = 1.017E14;
  k2_ = 13.21;
end

  environment: baylor_2002f__environment(time_);

  Ca2Trop: baylor_2002f__Ca2Trop(Ca2Trop0, k2, k2_, Ca0, CaTrop0, time_);
  reaction_constants: baylor_2002f__reaction_constants(k1, k1_, k2, k2_);

model baylor_model_f_2002__environment(time_)
end

model baylor_model_f_2002__Ca(Ca, k1, k1_, k2, k2_, Trop, CaTrop, Ca2Trop, time_)

  // Rate Rules:
  Ca' = k1_ * CaTrop + k2_ * Ca2Trop - ( k1 * Ca * Trop + k2 * Ca * CaTrop );

  // Variable initializations:
  Ca = 0.05;
end

model baylor_model_f_2002__Trop(Trop, k1, k1_, Ca, CaTrop, time_)

  // Rate Rules:
  Trop' = k1_ * CaTrop - k1 * Ca * Trop;

  // Variable initializations:
  Trop = 360.0;
end

model baylor_model_f_2002__CaTrop(CaTrop, k1, k1_, k2, k2_, Trop, Ca, Ca2Trop, time_)

  // Rate Rules:
  CaTrop' = k1 * Ca * Trop + k2_ * Ca2Trop - ( k1_ * CaTrop + k2 * Ca * CaTrop );
end

model baylor_model_f_2002__Ca2Trop(Ca2Trop, k2, k2_, Ca, CaTrop, time_)

  // Rate Rules:
  Ca2Trop' = k2 * Ca * CaTrop - k2_ * Ca2Trop;
end

model baylor_model_f_2002__reaction_constants(k1, k1_)

  // Variable initializations:
  k1 = 2.033E8;
  k1_ = 2642.0;
  k2 = 1.017E8;
  k2_ = 13.21;
end

model baylor_model_f_2002____main()

  // Sub-modules, and any changes to those submodules:
  environment: baylor_model_f_2002__environment(time_);
  Ca: baylor_model_f_2002__Ca(Ca0, k1, k1_, k2, k2_, Trop0, CaTrop0, Ca2Trop0, time_);
  Trop: baylor_model_f_2002__Trop(Trop0, k1, k1_, Ca0, CaTrop0, time_);
  CaTrop: baylor_model_f_2002__CaTrop(CaTrop0, k1, k1_, k2, k2_, Trop0, Ca0, Ca2Trop0, time_);
  Ca2Trop: baylor_model_f_2002__Ca2Trop(Ca2Trop0, k2, k2_, Ca0, CaTrop0, time_);
  reaction_constants: baylor_model_f_2002__reaction_constants(k1, k1_);
  reaction_constants.k2 is k2;
  reaction_constants.k2_ is k2_;
end

end

model devries_sherman_2001_version01__membrane(V, tau, time_, i_K, i_Ca, i_s)

  // Rate Rules:
  V' = -(( i_Ca + i_K + i_s )) / tau;

  // Variable initializations:
  tau = 20.0;
end

model devries_sherman_2001_version01__calcium_current_m_gate(m_infinity, V)

  // Assignment Rules:
  m_infinity := 1.0 / ( 1.0 + exp(( V_m - V ) / theta_m ));

  // Variable initializations:
  V_m = -20.0;
  theta_m = 12.0;
end

model devries_sherman_2001_version01__calcium_current(i_Ca, time_, V, m_infinity)

  // Sub-modules, and any changes to those submodules:
  calcium_current_m_gate: devries_sherman_2001_version01__calcium_current_m_gate(m_infinity, V);

  // Assignment Rules:
  i_Ca := g_Ca * m_infinity * ( V - V_Ca );

  // Variable initializations:
  g_Ca = 3.6;
  V_Ca = 25.0;
end

model devries_sherman_2001_version01__potassium_current_n_gate(n, V, tau, time_)

  // Assignment Rules:
  n_infinity := 1.0 / ( 1.0 + exp(( V_n - V ) / theta_n ));

  // Rate Rules:
  n' = lamda * ( n_infinity - n ) / tau;

  // Variable initializations:
  V_n = -16.0;
  theta_n = 5.6;
  lamda = 0.8;
end

model devries_sherman_2001_version01__potassium_current(i_K, V_K, time_, V, tau, n)

  // Sub-modules, and any changes to those submodules:
  potassium_current_n_gate: devries_sherman_2001_version01__potassium_current_n_gate(n, V, tau, time_);

  // Assignment Rules:
  i_K := g_K * n * ( V - V_K );

  // Variable initializations:
  V_K = -75.0;
  g_K = 10.0;
end

model devries_sherman_2001_version01__slow_current_s_gate(s, V, time_)

  // Assignment Rules:
  s_infinity := 1.0 / ( 1.0 + exp(( V_s - V ) / theta_s ));

  // Rate Rules:
  s' = ( s_infinity - s + beta ) / tau_s;

  // Variable initializations:
  V_s = -38.0;
  theta_s = 10.0;
  beta = 0.0;
  tau_s = 35000.0;
end

model devries_sherman_2001_version01__slow_current(i_s, time_, V, V_K, s)

  // Sub-modules, and any changes to those submodules:
  slow_current_s_gate: devries_sherman_2001_version01__slow_current_s_gate(s, V, time_);

  // Assignment Rules:
  i_s := g_s * s * ( V - V_K );

  // Variable initializations:
  g_s = 4.0;
end

  // Sub-modules, and any changes to those submodules:
  environment: devries_sherman_2001_version01__environment(time_);
  membrane: devries_sherman_2001_version01__membrane(V, tau, time_, i_K, i_Ca, i_s);
  calcium_current: devries_sherman_2001_version01__calcium_current(i_Ca, time_, V, m_infinity);
  potassium_current: devries_sherman_2001_version01__potassium_current(i_K, V_K, time_, V, tau, n);
  slow_current: devries_sherman_2001_version01__slow_current(i_s, time_, V, V_K, s);

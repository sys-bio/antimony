model autonomics_model__pressure_driving_autonomic_receptors(PA, PA1, CRRFLX, EXE)

  // Assignment Rules:
  PA1 := piecewise( CRRFLX , CRRFLX > 0.0000001 , PA - EXE );
end

model autonomics_model__chemoreceptors_effect_of_PA(PA1, AUC, AUC1)

  // Assignment Rules:
  AUC := piecewise( 0.005 * (80 - PA1) * AUC1 , ( PA1 < 80) && (PA1 >= 40 ), 0.2 * AUC1 , PA1 < 40 , 0 );
end

model autonomics_model__chemoreceptors_effect_of_art_PO2(PO2ART, AUC, AUC3, O2CHMO)

  // Assignment Rules:
  AUC3 := AUC + AUC2;
  AUC2 := piecewise( O2CHMO * (80 - PO2ART) , ( PO2ART < 80) && (PO2ART >= 40 ), O2CHMO * 40 , PO2ART < 40 , 0 );
end

model autonomics_model__arterial_baroreceptor_reflex(PA1, AU6_init, AU4_init, time_, AU6C, AUX, AUK, BAROTC)

  // Assignment Rules:
  AU6C := AU6;
  AUB := piecewise( 0.016667 * (160 - PA1) , ( PA1 < 160) && (PA1 >= 80 ), 1.3336 , PA1 < 80 , 0 );
  A1B := (AUB - 1 ) * AUX + 1;
  AU6A := A1B - AU4;

  // Rate Rules:
  AU6' = (AU6A - AU6) / BAROTC;

  // Variable initializations:
  AU4 = AU4_init;
  AU6 = AU6_init;
end

model autonomics_model__CNS_ischemic_reflex(PA1, AUN, AUN1)

  // Assignment Rules:
  AUN := piecewise( 0.04 * (40 - PA1) * AUN1 , PA1 < 40 , 0 );
end

model autonomics_model__autonomic_response_to_vasculature_pressure(PLA, PRA, PPA, AULP, AULPM)

  // Assignment Rules:
  AULP := (15 / (PLA + PRA + PPA) - 1 ) * AULPM + 1;
end

model autonomics_model__autonomic_response_to_exercise(AUEX, EXC, EXCXP)

  // Assignment Rules:
  AUEX := power(EXC, EXCXP);
end

model autonomics_model__total_autonomic_stimulation(AUC3, AU6C, AUN, AULP, AUEX, AUTTL, EXCML)

  // Assignment Rules:
  AUTTL := piecewise( 0 , AUTTL1 < 0 , AUTTL1 );
  AUTTL1 := (AUEX * AULP * (AUC3 + AU6C + AUN) - 1 ) * EXCML + 1;
end

model autonomics_model__actual_autonomic_stimulation(AUTTL, AU1_init, time_, AU, AUDMP, AUMAX, AUMIN, AUSLP)

  // Assignment Rules:
  AU := piecewise( AUMIN , AUT < AUMIN , AUT );
  DAU := (AUTTL - AU1) / AUDMP;
  AUT := AUMAX - (AUMAX - 1 ) / exp(AUSLP * (AU1 - 1 ));

  // Rate Rules:
  AU1' = DAU;

  // Variable initializations:
  AU1 = AU1_init;
end

model autonomics_model__autonomic_drive_on_target_organs_and_tissues(AU, VVR, AUH, AUR, AOM, AUM, AVE, VV9, AUL, AUV, AUS, O2A, AUM1, AUM2, AUY)

  // Assignment Rules:
  VVR := VV9 - AU * AUL + AUL;
  AUH := AUO * AUV + 1;
  AUR := AUO * AUS + 1;
  AOM := AUO * O2A + 1;
  AUM := power(AUO * AUM1 + 1 , AUM2);
  AVE := AUO * AUY + 1;
  AUO := AU - 1;
end

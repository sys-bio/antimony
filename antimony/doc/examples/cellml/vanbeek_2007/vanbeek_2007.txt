model vanbeek_model_2007__J_CKMM(J_CKMM, ATP_cyt, ADP_cyt, PCr_cyt, Cr_cyt)

  // Assignment Rules:
  J_CKMM := ( Vmax_MM_f * ( ATP_cyt * Cr_cyt / ( Kia * Kb )) - Vmax_MM_b * ( ADP_cyt * PCr_cyt / ( Kic * Kd ))) / Den_MMCK;
  Den_MMCK := 1.0 + Cr_cyt / Kib + PCr_cyt / Kid + ATP_cyt * ( 1.0 / Kia + Cr_cyt / ( Kia * Kb )) + ADP_cyt * ( 1.0 / Kic + PCr_cyt / ( Kid * Kc ) + Cr_cyt / ( Kic * KIb ));
  KIb := Kib;
  Kc := Kic * Kd / Kid;

  // Variable initializations:
  Kia = 9.0E2;
  Kb = 1.55E4;
  Kib = 3.49E4;
  Kic = 2.224E2;
  Kd = 1.67E3;
  Kid = 4.73E3;
  Vmax_MM_f = 1.144E4;
  Vmax_MM_b = 4.804E4;
end

model vanbeek_model_2007__J_CKMi(J_CKMi, ATP_ims, ADP_ims, PCr_ims, Cr_ims)

  // Assignment Rules:
  J_CKMi := ( Vmax_Mi_f * ( ATP_ims * Cr_ims / ( Kia * Kb )) - Vmax_Mi_b * ( ADP_ims * PCr_ims / ( Kic * Kd ))) / Den_MiCK;
  Den_MiCK := 1.0 + Cr_ims / Kib + PCr_ims / Kid + ATP_ims * ( 1.0 / Kia + Cr_ims / ( Kia * Kb )) + ADP_ims * ( 1.0 / Kic + PCr_ims / ( Kid * Kc ) + Cr_ims / ( Kic * KIb ));
  KIb := Kib;
  Kc := Kic * Kd / Kid;

  // Variable initializations:
  Kia = 7.5E2;
  Kb = 5.2E3;
  Kib = 2.88E4;
  Kic = 2.048E2;
  Kd = 5.0E2;
  Kid = 1.6E3;
  Vmax_Mi_f = 8.82E2;
  Vmax_Mi_b = 3.704E3;
end

model vanbeek_model_2007__J_hyd(J_hyd, time_)

  // Assignment Rules:
  J_hyd := piecewise( H_ATPmax * ltime / t_cycle * 6 , ( ltime >= 0) && (ltime < 1 / 6 * t_cycle ), H_ATPmax * (1 - 6 * (ltime / t_cycle - 1 / 6 )) , ( ltime >= 1 / 6 * t_cycle) && (ltime < 1 / 3 * t_cycle ), 0 , ( ltime >= 1 / 3 * t_cycle) && (ltime < t_cycle ) );
  H_ATPmax := piecewise( 6 * J_hyd_basis_1 , time_ <= duration_1 , 6 * J_hyd_basis_2 );
  t_cycle_1 := 60 / freq_1;
  t_cycle_2 := 60 / freq_2;
  t_cycle := piecewise( t_cycle_1 , time_ <= duration_1 , t_cycle_2 );
  duration_1 := nb_of_cycles_1 * t_cycle_1;
  ltime := piecewise( time_ - t_cycle_1 * floor(time_ / t_cycle_1) , time_ <= duration_1 , time_ - duration_1 - t_cycle_2 * floor((time_ - duration_1) / t_cycle_2) );

  // Variable initializations:
  J_hyd_basis_1 = 4.865e2;
  J_hyd_basis_2 = 6.276e2;
  freq_1 = 135;
  freq_2 = 220;
  nb_of_cycles_1 = 5;
end

model vanbeek_model_2007__J_syn(J_syn, Pi_ims, ADP_ims)

  // Assignment Rules:
  J_syn := V_max_syn * ( ADP_ims * Pi_ims / ( KPi * KADP * Den_syn ));
  Den_syn := 1.0 + ADP_ims / KADP + Pi_ims / KPi + ADP_ims * Pi_ims / ( KADP * KPi );

  // Variable initializations:
  KPi = 8E2;
  KADP = 25;
  V_max_syn = 1.504E4;
end

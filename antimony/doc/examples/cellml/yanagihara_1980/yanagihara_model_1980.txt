model yanagihara_model_1980__membrane(V, time_, i_Na, i_K, i_Leak, i_s, i_h)

  // Rate Rules:
  V' = -((i_Na + i_K + i_Leak + i_s + i_h)) / Cm;

  // Variable initializations:
  V = -62.83;
  Cm = 1;
end

model yanagihara_model_1980__sodium_channel_m_gate(m, V, time_)

  // Assignment Rules:
  alpha_m := 1 * (V + 37 ) / (-(exp((V + 37 ) / -(10 ))) + 1 );
  beta_m := 40 * exp((V + 62 ) / -(17.8 ));

  // Rate Rules:
  m' = alpha_m * (1 - m) - beta_m * m;

  // Variable initializations:
  m = 0.047938;
end

model yanagihara_model_1980__sodium_channel_h_gate(h, V, time_)

  // Assignment Rules:
  alpha_h := 1.209E-3 * exp((V + 20 ) / -(6.534 ));
  beta_h := 1 / (1 + exp((V + 30 ) / -(10 )));

  // Rate Rules:
  h' = alpha_h * (1 - h) - beta_h * h;

  // Variable initializations:
  h = 0.95994;
end

model yanagihara_model_1980__sodium_channel(i_Na, time_, V, m, h)

  // Sub-modules, and any changes to those submodules:

  // Assignment Rules:

end

  // Rate Rules:

model tolic_2000__environment(time_, Vp, Vi, Vg, E)

  // Variable initializations:
  Vp = 3;
  Vi = 11;
  Vg = 10;
  E = 0.2;
end

model tolic_2000__plasma_insulin(Ip, time_, Vp, Vi, Vg, Ii, G, E)

  // Assignment Rules:
  Ip_conc := Ip / Vp;
  f1_G := Rm / (1 + exp((C1 - G / Vg) / a1));

  // Rate Rules:
  Ip' = f1_G - (E * (Ip / Vp - Ii / Vi) + Ip / tp);

  // Variable initializations:
  Ip = 93.36441699;
  tp = 6;
  Rm = 210;
  C1 = 2000;
  a1 = 300;
end

model tolic_2000__intercellular_insulin(Ii, ti, time_, Vp, Vi, Ip, E)

  // Assignment Rules:
  Ii_conc := Ii / Vi;

  // Rate Rules:
  Ii' = E * (Ip / Vp - Ii / Vi) - Ii / ti;

  // Variable initializations:
  Ii = 243.2865183;
  ti = 100;
end

model tolic_2000__glucose(G, time_, ti, Vg, Vp, Vi, E, Ii, x3)

  // Assignment Rules:
  G_conc := G / (Vg * 10 );
  f2_G := Ub * (1 - exp(-(G) / (C2 * Vg)));
  f3_G := G / (C3 * Vg);
  f4_Ii := U0 + (Um - U0) / (1 + exp(-(beta) * ln(Ii / C4 * (1 / Vi + 1 / (E * ti)))));
  f5_x3 := Rg / (1 + exp(alpha * (x3 * 1 / Vp - C5)));

  // Rate Rules:
  G' = Gin + f5_x3 + -((f2_G + f3_G * f4_Ii));

  // Variable initializations:
  G = 12342.61665;
  Gin = 216;
  C2 = 144;
  C3 = 1000;
  C4 = 80;
  C5 = 26;
  U0 = 40;
  Um = 940;
  Ub = 72;
  beta = 1.77;
  Rg = 180;
  alpha = 0.29;
end

model tolic_2000__delay(x3, time_, Ip)

  // Rate Rules:
  x3' = 3 / td * (x2 - x3);
  x1' = 3 / td * (Ip / 1 - x1);
  x2' = 3 / td * (x1 - x2);

  // Variable initializations:
  x3 = 104.5878705;
  td = 36;
  x1 = 110.420253;
  x2 = 112.7601171;
end

model tolic_2000____main()

  // Sub-modules, and any changes to those submodules:
  environment: tolic_2000__environment(time_, Vp, Vi, Vg, E);
  plasma_insulin: tolic_2000__plasma_insulin(Ip, time_, Vp, Vi, Vg, Ii, G, E);
  intercellular_insulin: tolic_2000__intercellular_insulin(Ii, ti, time_, Vp, Vi, Ip, E);
  glucose: tolic_2000__glucose(G, time_, ti, Vg, Vp, Vi, E, Ii, x3);
  delay_: tolic_2000__delay(x3, time_, Ip);
end

model maldonado_2006__environment(time_)
end

model maldonado_2006__A_B(A_B, r_B)

  // Assignment Rules:
  A_B := power( r_B , 2 ) * pi;
end

model maldonado_2006__r_B(r_B, xc, xb, xy, time_)

  // Rate Rules:
  r_B' = k_for / X_b * xb + 1 / X_y * xy - ( k_res / X_c * xc + k_rB * r_B );

  // Variable initializations:
  r_B = 0.9912;
  X_c = 9.127e-4;
  X_b = 7.282e-4;
  X_y = 7.300e-3;
  k_rB = 1.0e0;
  k_for = 1.0e-3;
  k_res = 10.0e-3;
end

model maldonado_2006__xy(xy, xb, time_)

  // Rate Rules:
  xy' = k_byp * ( xb - X_bss ) - k_yd * ( xy - X_yss );

  // Variable initializations:
  xy = 7.300e-3;
  X_bss = 7.282e-4;
  X_yss = 7.300e-3;
  k_byp = 1.00e-1;
  k_yd = 1.00e0;
end

model maldonado_2006__F_sti(F_sti, xy, F_s)

  // Assignment Rules:
  F_sti := F_s * xy / ( 1 + exp(-(( k_Fs * F_s + k_y * xy ))));

  // Variable initializations:
  k_Fs = 1.00e0;
  k_y = 1.00e0;
end

model maldonado_2006__F_s(F_s, A_B, time_)

  // Assignment Rules:
  F_s := F_a / A_B;
  F_a := piecewise( 10000.0 , ( time_ >= 100.0) && (time_ < 130.0 ), 100.0 );
end

model maldonado_2006__x_no(x_no, F_sti, time_)

  // Rate Rules:
  x_no' = k_yno * F_sti + X_noe - k_nod * x_no;

  // Variable initializations:
  x_no = 0.02;
  X_noe = 0e0;
  k_yno = 2e4;
  k_nod = 1e3;
end

model maldonado_2006__x_pge(x_pge, F_sti, x_no, time_)

  // Rate Rules:
  x_pge' = k_ypge * F_sti + k_nopge * x_no + X_pgex - k_pged * x_pge;

  // Variable initializations:
  x_pge = 0.01;
  X_pgex = 0e0;
  k_pged = 1e2;
  k_nopge = 1e0;
  k_ypge = 1e2;
end

model maldonado_2006__x_opg(x_opg, Io, K_o_p, xr, x_no, pi_c, time_)

  // Rate Rules:
  x_opg' = K_o_p / 1 * pi_c * xr + Io + k_nopg * x_no - k_opgd * x_opg;

  // Variable initializations:
  x_opg = 0.01;
  k_opgd = 3.5e-1;
  k_nopg = 1e1;
end

model maldonado_2006__x_kl(x_kl, pi_L, Io, K_o_p, xb, xr, x_no, x_opg, pi_p, time_)

  // Assignment Rules:
  pi_L := k3 / k4 * K_l_p * pi_p * xb / (1 + k3 * K / k4 + k1 / (k2 * ko) * (K_o_p / 1 / pi_p * xr + Io)) * (1 + Il / rl);

  // Rate Rules:
  x_kl' = rl + Il - ( k_nokl * x_no + rl * (( 1 + k1 / k2 * x_opg + k3 / k4 * K ) / ( K_l_p * pi_p * xb )) * x_kl );

  // Variable initializations:
  x_kl = 0.01;
  K_l_p = 3e6;
  K = 10;
  k_nokl = 1e2;
  Il = 0;
  rl = 1e3;
  k1 = 1e-2;
  k2 = 10;
  k3 = 5.8e-4;
  k4 = 1.7e-2;
  ko = 0.35;
end

model maldonado_2006__xr(xr, x_pge, D_B, pi_c, time_)

  // Rate Rules:
  xr' = D_R * pi_c + k_pger * x_pge - D_B / pi_c * xr;

  // Variable initializations:
  xr = 7.734e-4;
  D_R = 7e-4;
  k_pger = 1e-4;
end

model maldonado_2006__xb(xb, xr, D_B, pi_c, time_)

  // Rate Rules:
  xb' = D_B / pi_c * xr - k_B * xb;

  // Variable initializations:
  xb = 7.282e-4;
  k_B = 0.189;
end

model maldonado_2006__xc(xc, pi_c, pi_L, time_)

  // Rate Rules:
  xc' = D_C * pi_L - D_A * pi_c * xc;

  // Variable initializations:
  xc = 9.127e-4;
  D_C = 0.189;
  D_A = 0.7;
end

model maldonado_2006__model_parameters(D_B, pi_c, pi_p, K_o_p, Io, xc)

  // Assignment Rules:
  D_B := f0 * dB;
  pi_c := (xc + f0 * C_s) / (xc + C_s);
  pi_p := (P + P_0) / (P + P_s);
  P := IP / kP;
  P_0 := SP / kP;
  P_s := k6 / k5;

  // Variable initializations:
  K_o_p = 2e5;
  Io = 0.0;
  f0 = 0.05;
  dB = 0.7;
  C_s = 5e-3;
  SP = 250;
  k5 = 0.02;
  k6 = 3;
  IP = 0;
  kP = 86;
end

  environment: maldonado_2006__environment(time_);

  xy: maldonado_2006__xy(xy0, xb0, time_);

  F_s: maldonado_2006__F_s(F_s0, A_B0, time_);

  x_pge: maldonado_2006__x_pge(x_pge0, F_sti0, x_no0, time_);
  x_opg: maldonado_2006__x_opg(x_opg0, Io, K_o_p, xr0, x_no0, pi_c, time_);
  x_kl: maldonado_2006__x_kl(x_kl0, pi_L, Io, K_o_p, xb0, xr0, x_no0, x_opg0, pi_p, time_);

end

model cloutier_model_2009__Vg_gs(Vg_gs, Km_GLU, Km_ATP, GLUg, ATPg)

  Vg_gs := Vmax_g_gs * (GLUg / (GLUg + Km_GLU) * (ATPg / (ATPg + Km_ATP)));

model cloutier_model_2009__Vc_CO2(Vc_CO2, Fin_t, Vc, CO2a, CO2c)

model cloutier_model_2009__Vg_glys(Vg_glys, G6Pg, GLYg, GLY_inh, aGLY_inh)

  // Assignment Rules:
  Vg_glys := Vmax_glys * ( G6Pg / ( G6Pg + Km_G6P_glys )) * ( 1 - 1 / ( 1 + exp(-( aGLY_inh ) * ( 1 * ( GLYg - GLY_inh )))));

  // Assignment Rules:

  deltaVt_GLY := 1 + stim * (delta_GLY * KO3 * (1 / (1 + exp(1 * -(sr_GLY) * (time_ - (to + to_GLY))))) * (1 - unitstepSB2));

  Km_GLY = 1.0;
end

model cloutier_model_2009__Fin_t(Fin_t, stim, to, tend, sr, t1, deltaf, time_)

  // Assignment Rules:
  Fin_t := CBF0 + ( stim * CBF0 * deltaf * ( 1 / ( 1 + exp( 1 * -( sr ) * ( time_ - ( to + t1 - 3 ))))) - stim * CBF0 * deltaf * ( 1 / ( 1 + exp( 1 * -( sr ) * ( time_ - ( to + tend + t1 + 3 ))))));

end

model cloutier_model_2009__Fout_t(Fout_t, Fin_t, CBF0, Vv, Vv0, tv)

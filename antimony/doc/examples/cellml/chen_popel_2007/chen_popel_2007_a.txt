model chen_popel_model_2006_version01__Arg(Arg, Fe3, Fe3_Arg, Fe2_Arg, Fe2, k1, k_1, k4, k_4, time_)

  // Assignment Rules:
  S := k1 * Arg * Fe3 + k4 * Arg * Fe2 - ( k_1 * Fe3_Arg + k_4 * Fe2_Arg );

  // Rate Rules:
  Arg' = k_1 * Fe3_Arg + k_4 * Fe2_Arg + S - ( k1 * Arg * Fe3 + k4 * Arg * Fe2 );

  // Variable initializations:
  Arg = 100.0;
end

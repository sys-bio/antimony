  x' = lamda - ( x + beta * x * vI );
end

  // Rate Rules:
  y' = beta * ( time_ - tau ) * x * ( time_ - tau ) * vI * ( time_ - tau ) * exp( a_ * tau ) - a * y;

  // Variable initializations:
  tau = 12.0;
  a = 0.5;
end

end

end

model smith_1980__R(R, T, time_)

  // Assignment Rules:
  H := piecewise( 1 , x > 0 , 0 );
  x := T - c / h;

  // Rate Rules:
  R' = (c - h * T) * (1 - H) - b1 * R;

  // Variable initializations:
  R = 0.5;
  h = 12;
  c = 100;
  b1 = 1.29;
end

model smith_1980__L(L, R, time_)

  // Rate Rules:
  L' = g1 * R - b2 * L;

  // Variable initializations:
  L = 22;
  g1 = 10;
  b2 = 0.97;
end

model smith_1980__T(T, L, time_)

  // Rate Rules:
  T' = g2 * L - b3 * T;

  // Variable initializations:
  T = 15;
  g2 = 0.7;
  b3 = 1.39;
end

model smith_1980____main()

  // Sub-modules, and any changes to those submodules:
  environment: smith_1980__environment(time_);
  R: smith_1980__R(R0, T0, time_);
  L: smith_1980__L(L0, R0, time_);
  T: smith_1980__T(T0, L0, time_);
end

model lenbury_1991__x(x, b, z, y, time_)

  // Rate Rules:
  x' = alpha * exp(a * (1 - power(z, 2 )) + b * (1 - power(y, 2 ))) + D * cos(omega * time_) - alpha * x;

  // Variable initializations:
  x = 1.5;
  alpha = 0.5;
  omega = 2;
  D = 0.8228;
  a = 8.1252;
end

model lenbury_1991__y(y, b, z, x, time_)

  // Rate Rules:
  y' = beta * x * exp(b * (1 - power(z, 2 ))) - beta * y;

  // Variable initializations:
  y = 0.95;
  beta = 0.38;
end

model lenbury_1991__z(z, y, time_)

  // Rate Rules:
  z' = gamma * y - gamma * z;

  // Variable initializations:
  z = 1.0;
  gamma = 0.6;
end

  x' = alpha * exp( a * ( 1.0 - power( z , 2.0 )) + b * ( 1.0 - power( y , 2.0 ))) + D * u - alpha * x;

  // Variable initializations:
  alpha = 0.5;
  D = 0.8228;
  a = 8.1252;
end

  // Rate Rules:
  y' = beta * x * exp( b * ( 1.0 - power( z , 2.0 ))) - beta * y;

  // Variable initializations:
  beta = 0.38;
end

  // Rate Rules:
  z' = gamma * y - gamma * z;

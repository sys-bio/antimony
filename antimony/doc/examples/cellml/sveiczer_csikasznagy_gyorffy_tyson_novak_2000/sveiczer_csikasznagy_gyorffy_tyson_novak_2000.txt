model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__MPF(MPF, k1, k2, kwee, kc25, kj, kjr, k6_, k6, mass, preMPF, Rum1, Rum1P, CR, CRP, time_)

  // Rate Rules:
  MPF' = k1 * mass - k2 * MPF - kwee * MPF + kc25 * preMPF - kj * MPF * (Rum1 + Rum1P) + kjr * (CR + CRP) + k6_ * CR + (k6_ + k6) * CRP;

  // Variable initializations:
  MPF = 0;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__preMPF(preMPF, kwee, k2, kc25, MPF, time_)

  // Rate Rules:
  preMPF' = kwee * MPF - kc25 * preMPF - k2 * preMPF;

  // Variable initializations:
  preMPF = 1;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Ste9(Ste9, kste9r_, kste9r, kste9, MPF_a, PP, SK, mass, time_)

  // Rate Rules:
  Ste9' = (kste9r_ + kste9r * PP) * (1 - Ste9) / (Jste9r + 1 - Ste9) - kste9 * (MPF_a + SK * mass) * Ste9 / (Jste9 + Ste9);

  // Variable initializations:
  Ste9 = 0;
  Jste9r = 0.01;
  Jste9 = 0.01;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Mik1(Mik1, ks, kmr_, kmr, km, MPF_a, PP, time_)

  // Rate Rules:
  Mik1' = (ks + kmr_ + kmr * PP) * (1 - Mik1) / (Jmikr + 1 - Mik1) - km * MPF_a * Mik1 / (Jmik + Mik1);

  // Variable initializations:
  Mik1 = 0;
  Jmikr = 0.15;
  Jmik = 0.15;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Wee1(Wee1, kwr_, kwr, kw, MPF_a, PP, time_)

  // Rate Rules:
  Wee1' = (kwr_ + kwr * PP) * (1 - Wee1) / (Jweer + 1 - Wee1) - kw * MPF_a * Wee1 / (Jwee + Wee1);

  // Variable initializations:
  Wee1 = 0;
  Jweer = 0.2;
  Jwee = 0.2;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Cdc25(Cdc25, k25, k5, k25r_, k25r, MPF_a, PP, time_)

  // Rate Rules:
  Cdc25' = k25 * MPF_a * (1 - Cdc25) / (J25 + 1 - Cdc25) - (k5 + k25r_ + k25r * PP) * Cdc25 / (J25r + Cdc25);

  // Variable initializations:
  Cdc25 = 0;
  J25 = 0.05;
  J25r = 0.05;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Slp1(Slp1, kas, kad, MPF_a, time_)

  // Rate Rules:
  Slp1' = kas * MPF_a - kad * Slp1;

  // Variable initializations:
  Slp1 = 0;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Slp1_a(Slp1_a, kaa_, kaa, kai, kad, MPF_a, Slp1, time_)

  // Rate Rules:
  Slp1_a' = (kaa_ + kaa * MPF_a) * (Slp1 - Slp1_a) - (kai + kad) * Slp1_a;

  // Variable initializations:
  Slp1_a = 0;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Inh(Inh, k3, ki, kir, k4, PP, PI_, time_)

  // Rate Rules:
  Inh' = k3 - ki * Inh * PP + kir * PI_ - k4 * Inh;

  // Variable initializations:
  Inh = 0;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__PI(PI_, ki, kir, k4, Inh, PP, time_)

  // Rate Rules:
  PI_' = ki * Inh * PP - kir * PI_ - k4 * PI_;

  // Variable initializations:
  PI_ = 0.2;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__CRP(CRP, kp, kpp_, kpp, kj, kjr, k2c, k6_, k6, epsilon_p, MPF_a, SK, mass, PP, CR, MPF, Rum1P, time_)

  // Rate Rules:
  CRP' = kp * (MPF_a + epsilon_p * SK * mass) * CR - (kpp_ + kpp * PP) * CRP + kj * MPF * Rum1P - kjr * CRP - k2c * CRP - (k6_ + k6) * CRP;

  // Variable initializations:
  CRP = 0;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__CR(CR, kj, kjr, k2c, k6_, kp, kpp_, kpp, epsilon_p, Rum1, MPF, MPF_a, SK, mass, PP, CRP, time_)

  // Rate Rules:
  CR' = kj * MPF * Rum1 - kjr * CR - k2c * CR - k6_ * CR - kp * (MPF_a + epsilon_p * SK * mass) * CR + (kpp_ + kpp * PP) * CRP;

  // Variable initializations:
  CR = 0;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Rum1(Rum1, k5, k6_, kp, kpp_, kpp, kj, kjr, k2c, epsilon_p, MPF_a, SK, mass, PP, Rum1P, MPF, CR, time_)

  // Rate Rules:
  Rum1' = k5 - k6_ * Rum1 - kp * (MPF_a + epsilon_p * SK * mass) * Rum1 + (kpp_ + kpp * PP) * Rum1P - kj * MPF * Rum1 + kjr * CR + k2c * CR;

  // Variable initializations:
  Rum1 = 0;
end

model sveiczer_csikasznagy_gyorffy_tyson_novak_2000_version01__Rum1P(Rum1P, kp, kpp_, kpp, k6_, k6, kj, kjr, k2c, epsilon_p, SK, mass, Rum1, PP, MPF_a, MPF, CRP, time_)

  // Rate Rules:
  Rum1P' = kp * (MPF_a + epsilon_p * SK * mass) * Rum1 - (kpp_ + kpp * PP) * Rum1P - (k6_ + k6) * Rum1P - kj * MPF * Rum1P + kjr * CRP + k2c * CRP;

  // Variable initializations:
  Rum1P = 0;
end

model cui_kaandorp_lloyd_a_2008_version01__Py(Py, Zn, Py1, r3, r4, time_)

  // Rate Rules:
  Py' = r4 * Py1 - r3 * power( Zn , 2 ) * Py;

  // Variable initializations:
  Py = 25.0;
end

  Py1' = r3 * power( Zn , 2 ) * Py + k_1 * Qw2 - ( r4 * Py1 + k1a * Dw * Py1 );

  // Variable initializations:

end

model cui_kaandorp_lloyd_a_2008_version01__Dw(Dw, Qw2, Py1, Qw1, Rw, k1a, k_1, k2, k_2, k3, time_)

end

model brown_model_1997__environment(time_)
end

model brown_model_1997__H1(H1, A, beta_I, time_)

  // Rate Rules:
  H1' = -(( beta_I * H1 )) + A;

  // Variable initializations:
  H1 = 0.05;
end

model brown_model_1997__H2(H2, H1, beta_I, beta_C, time_)

  // Rate Rules:
  H2' = beta_I * H1 - beta_C * H2;

  // Variable initializations:
  H2 = 1.0;
end

model brown_model_1997__A(A, alpha, lamda, time_)

  // Assignment Rules:
  A := piecewise( A_max * (( 1.0 - exp(-( lamda ) * ( time_ - t_on ))) / ( 1.0 - exp(-( lamda ) * ( t_off - t_on )))) , ( time_ < t_off) && (time_ >= t_on ), A_max * exp(-( alpha ) * ( time_ - t_off )) , time_ >= t_off , 0.0 );

  // Variable initializations:
  t_on = 1316.0;
  t_off = 1792.0;
  A_max = 6.51;
end

model brown_model_1997__model_parameters(beta_I, beta_C, alpha, lamda)

  // Assignment Rules:
  beta_I := ln( 2.0 ) / tau_I;
  beta_C := ln( 2.0 ) / tau_C;
  alpha := ln( 2.0 ) / tau_alpha;
  lamda := ln( 2.0 ) / tau_lamda;

  // Variable initializations:
  tau_I = 2.82;
  tau_C = 23.67;
  tau_alpha = 25.92;
  tau_lamda = 24.04;
end

model brown_model_1997____main()

  // Sub-modules, and any changes to those submodules:
  environment: brown_model_1997__environment(time_);
  H1: brown_model_1997__H1(H10, A0, beta_I, time_);
  H2: brown_model_1997__H2(H20, H10, beta_I, beta_C, time_);
  A: brown_model_1997__A(A0, alpha, lamda, time_);
  model_parameters: brown_model_1997__model_parameters(beta_I, beta_C, alpha, lamda);
end

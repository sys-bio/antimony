model dash_bassingthwaighte_model_2004_version01__SHbO2(SHbO2, KHbO2, O2)

  // Assignment Rules:
  SHbO2 := KHbO2 * O2 / ( 1.0 + KHbO2 * O2 );
end

model dash_bassingthwaighte_model_2004_version01__SHbCO2(SHbCO2, KHbCO2, CO2)

  // Assignment Rules:
  SHbCO2 := KHbCO2 * CO2 / ( 1.0 + KHbCO2 * CO2 );
end

model dash_bassingthwaighte_model_2004_version01__KHbO2(KHbO2, CO2, Hrbc, K2, K2_, K3, K3_, K4, K5_, K6_)

  // Assignment Rules:
  KHbO2 := K4 * ( K3 * CO2 * ( 1.0 + K3_ / Hrbc ) + ( 1.0 + Hrbc / K6_ )) / ( K2 * CO2 * ( 1.0 + K2_ / Hrbc ) + ( 1.0 + Hrbc / K5_ ));
end

model dash_bassingthwaighte_model_2004_version01__KHbCO2(KHbCO2, O2, Hrbc, K2, K2_, K3, K3_, K4, K5_, K6_)

  // Assignment Rules:
  KHbCO2 := ( K2 * ( 1.0 + K2_ / Hrbc ) + K3 * K4 * ( 1.0 + K3_ / Hrbc ) * O2 ) / ( 1.0 + Hrbc / K5_ + K4 * ( 1.0 + Hrbc / K6_ ) * O2 );
end

model dash_bassingthwaighte_model_2004_version01__K4(K4, O2_S, H_S, n1, n2, CO2_S, K4_, n0, O2, CO2, Hrbc)

  // Assignment Rules:
  K4 := K4_ * power( O2 / O2_S , n0 ) * power( Hrbc / H_S , -( n1 )) * power( CO2 / CO2_S , -( n2 ));
end

model dash_bassingthwaighte_model_2004_version01__O2(O2, alpha_O2, PO2)

  // Assignment Rules:
  O2 := alpha_O2 * PO2;
end

model dash_bassingthwaighte_model_2004_version01__CO2(CO2, alpha_CO2, PCO2)

  // Assignment Rules:
  CO2 := alpha_CO2 * PCO2;
end

model dash_bassingthwaighte_model_2004_version01__alpha_O2(alpha_O2, Wpl, T)

  // Assignment Rules:
  alpha_O2 := ( 1.37 - 0.0137 * ( T - 37.0 ) + 0.00058 * power( T - 37.0 , 2.0 )) * ( 1E-6 / Wpl );
end

model dash_bassingthwaighte_model_2004_version01__alpha_CO2(alpha_CO2, Wpl, T)

  // Assignment Rules:
  alpha_CO2 := ( 3.07 - 0.057 * ( T - 37.0 ) + 0.002 * power( T - 37.0 , 2.0 )) * ( 1E-5 / Wpl );
end

model stern_1999__R(R, ki, ko, kim, kom, RI, O, Ca, time_)

  // Rate Rules:
  R' = kom * O + kim * RI - (ko * power(Ca, 2 ) * R + ki * Ca * R);

  // Variable initializations:
  R = 1;
end

model stern_1999__O(O, ki, ko, kim, kom, I, R, Ca, time_)

  // Rate Rules:
  O' = kim * I + ko * power(Ca, 2 ) * R - (kom * O + ki * Ca * O);

  // Variable initializations:
  O = 1;
end

model stern_1999__I(I, ki, ko, kim, kom, O, RI, Ca, time_)

  // Rate Rules:
  I' = ki * Ca * O + ko * power(Ca, 2 ) * RI - (kim * I + kom * I);

  // Variable initializations:
  I = 1;
end

model stern_1999__RI(RI, ki, ko, kim, kom, I, R, Ca, time_)

  // Rate Rules:
  RI' = ki * Ca * R + kom * I - (kim * RI + ko * power(Ca, 2 ) * RI);

  // Variable initializations:
  RI = 1;
end

model stern_1999__reaction_constants(Ca, ki, ko, kim, kom)

  // Variable initializations:
  Ca = 0.0001;
  ki = 0.5;
  ko = 35;
  kim = 0.005;
  kom = 0.06;
end

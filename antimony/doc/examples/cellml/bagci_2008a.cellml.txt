model cellmlmod_NO()

  // Assignment Rules:
  r_1NO := k_1NO
  r_4NO := k_4NO * NO * O_2m
  r_12aNO := k_12aNO * power(NO, 2 ) * O_2
  r_12bNOp := k_12bNOp * NO_2 * NO
  r_12bNOm := k_12bNOm * N2O3
  r_14NO := k_14NO * GSNO
  r_15NO := k_15NO * CcOX * NO
  r_16NO := k_16NO * FeLn * NO

  // Rate Rules:
  NO' = r_1NO - r_4NO - 2  * r_12aNO - r_12bNOp + r_12bNOm + r_14NO - r_15NO - r_16NO
  NO_2' = 2  * r_12aNO - r_12bNOp + r_12bNOm
  CcOX' = -(r_15NO)

  // Variable initializations:
  NO = 0;
  O_2 = 35;
  NO_2 = 0;
  CcOX = 0.1;

  //Other declarations:
  var NO, O_2m, O_2, NO_2, N2O3, GSNO, CcOX, FeLn, r_1NO, r_4NO, r_12aNO;
  var r_12bNOp, r_12bNOm, r_14NO, r_15NO, r_16NO, k_1NO, k_4NO, k_12aNO, k_12bNOp;
  var k_12bNOm, k_14NO, k_15NO, k_16NO, time;
end

model cellmlmod_O_2m()

  // Assignment Rules:
  r_2NO := k_2NO
  r_5NO := k_5NO * SOD * O_2m
  r_10NO := k_10NO * power(GSNO, 2 ) * O_2m

  // Rate Rules:
  O_2m' = r_2NO - r_4NO - r_5NO - r_10NO

  // Variable initializations:
  O_2m = 0;
  SOD = 10;

  //Other declarations:
  var O_2m, SOD, GSNO, r_2NO, r_4NO, r_5NO, r_10NO, k_2NO, k_5NO, k_10NO;
  var time;
end

model cellmlmod_ONOO_m()

  // Assignment Rules:
  r_6NO := k_6NO * ONOO_m * GSH
  r_7NO := k_7NO * ONOO_m * GPX
  r_8NO := k_8NO * ONOO_m * CO_2
  r_9NO := k_9NO * ONOO_m * Cyt_c

  // Rate Rules:
  ONOO_m' = r_4NO - r_6NO - r_7NO - r_8NO - r_9NO

  // Variable initializations:
  ONOO_m = 0;
  GPX = 5.8;
  CO_2 = 1000;
  Cyt_c = 400;

  //Other declarations:
  var ONOO_m, GSH, GPX, CO_2, Cyt_c, r_4NO, r_6NO, r_7NO, r_8NO, r_9NO, k_6NO;
  var k_7NO, k_8NO, k_9NO, time;
end

model cellmlmod_GSH()

  // Assignment Rules:
  GSSG := (GSH_0 - GSH - GSNO) / 2
  FeLnNO := FeLn_0 - FeLn
  r_11NO := k_11NO * N2O3 * GSH
  r_m := v_m * GSSG / (k_m + GSSG)
  r_17NO := k_17NO * FeLnNO * GSH

  // Rate Rules:
  GSH' = -(r_6NO) - r_11NO + 2  * r_m - r_17NO

  // Variable initializations:
  GSH = 10000;

  //Other declarations:
  var GSH, N2O3, GSSG, FeLnNO, FeLn, FeLn_0, GSH_0, GSNO, r_6NO, r_11NO, r_m;
  var r_17NO, k_11NO, v_m, k_m, k_17NO, time;
end

model cellmlmod_GSNO()

  // Rate Rules:
  GSNO' = r_6NO - 2  * r_10NO + r_11NO - r_14NO + r_17NO

  // Variable initializations:
  GSNO = 0;

  //Other declarations:
  var GSNO, r_6NO, r_10NO, r_11NO, r_14NO, r_17NO, time;
end

model cellmlmod_N2O3()

  // Assignment Rules:
  r_13NO := k_13NO * N2O3

  // Rate Rules:
  N2O3' = -(r_11NO) + r_12bNOp - r_12bNOm - r_13NO

  // Variable initializations:
  N2O3 = 0;

  //Other declarations:
  var N2O3, r_11NO, r_12bNOp, r_12bNOm, r_13NO, k_13NO, time;
end

model cellmlmod_FeLn()

  // Rate Rules:
  FeLn' = -(r_16NO) + r_17NO

  // Variable initializations:
  FeLn = 0.05;

  //Other declarations:
  var FeLn, r_16NO, r_17NO, time;
end

model nowak_1996a__uninfected(time_, beta, v, x)

  // Rate Rules:
  x' = lambda_ - d * x - beta * x * v;

  // Variable initializations:
  lambda_ = 1;
  d = 0.01;
  x = 100;
end

model nowak_1996a__infected(time_, beta, x, v, y)

  // Rate Rules:
  y' = beta * x * v - a * y;

  // Variable initializations:
  beta = 0.05;
  a = 0.5;
  y = 0;
end

model nowak_1996a__virus(time_, y, v)

  // Rate Rules:
  v' = k * y - u * v;

  // Variable initializations:
  k = 1;
  u = 1;
  v = 5;
end

  // Sub-modules, and any changes to those submodules:

  virus: nowak_1996a__virus(time_, y, v);

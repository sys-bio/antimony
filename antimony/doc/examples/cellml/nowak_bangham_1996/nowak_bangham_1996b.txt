model nowak_1996b__uninfected(time_, beta, v, x)

  // Rate Rules:
  x' = lambda_ - d * x - beta * x * v;

  // Variable initializations:
  lambda_ = 1;
  d = 0.01;
  x = 100;
end

model nowak_1996b__infected(time_, beta, x, v, z, y)

  // Rate Rules:
  y' = beta * x * v - a * y - p * y * z;

  // Variable initializations:
  beta = 0.02;
  a = 0.5;
  p = 1;
  y = 0;
end

model nowak_1996b__virus(time_, y, v)

  // Rate Rules:
  v' = k * y - u * v;

  // Variable initializations:
  k = 1;
  u = 1;
  v = 10;
end

model nowak_1996b__CTL(time_, y, z)

  // Rate Rules:
  z' = c * y * z - b * z;

  // Variable initializations:
  c = 1;
  b = 0.05;
  z = 1;
end

  // Sub-modules, and any changes to those submodules:
  environment: nowak_1996b__environment(time_);
  uninfected: nowak_1996b__uninfected(time_, beta, v, x);

  CTL: nowak_1996b__CTL(time_, y, z);

model kesmir_1999__B0(B0, pr, mu, rho, delta_B, CT_star, time_)

  // Rate Rules:
  B0' = pr * mu * CT_star - (rho * B0 + delta_B * B0);

  // Variable initializations:
  B0 = 3;
end

model kesmir_1999__B1(B1, alpha_B, rho, delta_B, B0, time_)

  // Rate Rules:

  // Variable initializations:

model kesmir_1999__centroblasts_sum(B_sum, B10, B9, B8, B7, B6, B5, B4, B3, B2, B1, B0)

  // Assignment Rules:
  B_sum := B1 + B1 + B3 + B4 + B5 + B6 + B6 + B7 + B8 + B9 + B10;
end

model kesmir_1999__C(C, rho, mu, B10, time_)

  // Rate Rules:
  C' = d * rho * B10 * 1 - mu * C;

  // Variable initializations:
  C = 0;
  d = 2;
end

model kesmir_1999__C_star(C_star, mu, CA, time_)

  // Rate Rules:
  C_star' = mu * CA - mu * C_star;

  // Variable initializations:
  C_star = 0;
end

model kesmir_1999__centrocytes_sum(C_starsum, C, C_star, CT_star, CA)

  // Assignment Rules:
  C_starsum := C + C_star;
end

model kesmir_1999__M(M, pr, mu, CT_star, time_)

  // Rate Rules:
  M' = (1 - pr) * mu * CT_star;

  // Variable initializations:
  M = 0;
end

model kesmir_1999__A(A, CA, time_)

  // Assignment Rules:
  log_A := log(A / 1 );

  // Rate Rules:
  A' = -(z) * A - u * CA * 1;

  // Variable initializations:
  A = 500;
  z = 0.02;
  u = 0.15;
end

model kesmir_1999__T(T, alpha_T, CT_star, time_)

  // Rate Rules:
  T' = sigma * 1 + p * alpha_T * CT_star - delta_T * T;

  // Variable initializations:
  T = 0;
  p = 2;
  sigma = 5;
  delta_T = 0.8;
end

model kesmir_1999__CA(CA, A, C)

  // Assignment Rules:
  CA := C * A / (SA * 1 + A);

  // Variable initializations:
  SA = 500;
end

model kesmir_1999__CT_star(CT_star, T, C_star, time_)

  // Assignment Rules:
  CT_star := C_star * T / (ST * 1 + C_star);

  // Variable initializations:
  ST = 50;
end

model kesmir_1999__alpha_B(alpha_B, B_sum, B0)

  // Assignment Rules:
  alpha_B := KB / (KB + B_sum / 1 );

  // Variable initializations:
  KB = 1e4;
end

model kesmir_1999__alpha_T(alpha_T, T)

  // Assignment Rules:
  alpha_T := KT / (KT + T / 1 );

  // Variable initializations:
  KT = 100;
end

model kesmir_1999__total(B_sum, C_starsum)

  // Assignment Rules:
  total := B_sum + C_starsum;
  log_total := log(total / 1 + 1E-12 );
end

model kesmir_1999__kinetic_parameters(rho, mu, delta_B, pr)

  // Variable initializations:
  rho = 4;
  mu = 3;
  delta_B = 0.8;
  pr = 0.15;
end

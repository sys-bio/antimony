model saftenku_2001__C1(C1, C1C2, C2C1, C2, Ca, time_)

  // Rate Rules:
  C1' = C2C1 * C2 - C1C2 * Ca * C2;

  // Variable initializations:
  C1 = 0.125;
end

model saftenku_2001__C2(C2, C1C2, C2C1, C2C3, C3C2, C2C5, C5C2, C1, C3, C5, Ca, time_)

  // Rate Rules:
  C2' = C1C2 * Ca * C1 + C3C2 * C3 + C5C2 * C5 - ( C2C1 * C2 + C2C3 * Ca * C2 + C2C5 * C2 );

  // Variable initializations:
  C2 = 0.125;
end

model saftenku_2001__C3(C3, O1C3, C3O1, C2C3, C3C2, O2C3, C3O2, O3C3, C3O3, C2, O1, O2, O3, Ca, time_)

  // Rate Rules:
  C3' = C2C3 * Ca * C2 + O1C3 * O1 + O3C3 * O3 + O2C3 * O2 - ( C3C2 * C3 + C3O1 * C3 + C3O2 * C3 + C3O3 * C3 );

  // Variable initializations:
  C3 = 0.125;
end

model saftenku_2001__C4(C4, O2C4, C4O2, O3C4, C4O3, O2, O3, time_)

  // Rate Rules:
  C4' = O2C4 * O2 + O3C4 * O3 - ( C4O2 * C4 + C4O3 * C4 );

  // Variable initializations:
  C4 = 0.125;
end

model saftenku_2001__C5(C5, C2C5, C5C2, C2, time_)

  // Rate Rules:
  C5' = C2C5 * C2 - C5C2 * C5;

  // Variable initializations:
  C5 = 0.125;
end

model saftenku_2001__O1(O1, O1C3, C3O1, C3, time_)

  // Rate Rules:
  O1' = C3O1 * C3 - O1C3 * O1;

  // Variable initializations:
  O1 = 0.125;
end

model saftenku_2001__O2(O2, C3O2, O2C3, O2C4, C4O2, C3, C4, time_)

  // Rate Rules:

end

  O3' = C3O3 * C3 + C4O3 * C4 - ( O3C3 * O3 + O3C4 * O3 );

  // Variable initializations:

end

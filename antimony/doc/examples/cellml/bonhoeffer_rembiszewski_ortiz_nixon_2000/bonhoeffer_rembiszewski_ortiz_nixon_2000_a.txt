model bonhoeffer_1990__T(T, b, I, time_)

  // Rate Rules:
  T' = s - ( dT * T + b * T * I );

  // Variable initializations:
  T = 1.0;
  s = 10.0;
  dT = 0.01;
end

model bonhoeffer_1990__I(I, b, T, E, time_)

  // Rate Rules:
  I' = b * T * I - ( dI * I + p * I * E );

  // Variable initializations:
  I = 1.0;
  p = 0.05;
  dI = 0.3;
end

model bonhoeffer_1990__E(E, I, time_)

  // Rate Rules:
  E' = c * I / (( I + K ) * E ) - dE * E;

  // Variable initializations:
  E = 1.0;
  c = 0.05;
  dE = 0.1;
  K = 0.1;
end

model bonhoeffer_1990__kinetic_parameters(b)

  // Variable initializations:
  b = 0.001;
end

model bonhoeffer_1990____main()

  // Sub-modules, and any changes to those submodules:
  environment: bonhoeffer_1990__environment(time_);
  T: bonhoeffer_1990__T(T0, b, I0, time_);

  E: bonhoeffer_1990__E(E0, I0, time_);
  kinetic_parameters: bonhoeffer_1990__kinetic_parameters(b);
end

model bonhoeffer_1990__T(T, b, I, time_)

  // Rate Rules:
  T' = s - ( dT * T + b * T * I );

  // Variable initializations:
  T = 1.0;
  s = 10.0;
  dT = 0.01;
end

model bonhoeffer_1990__I(I, b, ql, qa, T, E, time_)

  // Rate Rules:
  I' = b * T * I - ( dI * I + p * E * I + ql * I ) + qa * I;

  // Variable initializations:
  I = 1.0;
  p = 0.05;
  dI = 0.3;
end

model bonhoeffer_1990__Il(Il, ql, qa, I, time_)

  // Rate Rules:
  Il' = ql * I - ( al * Il + qa * Il );

  // Variable initializations:
  Il = 1.0;
  al = 0.01;
end

model bonhoeffer_1990__E(E, I, time_)

  // Rate Rules:
  E' = c * E * I / ( I + K ) - ( dE * E + dE_ * E * I / ( K_ + I ));

  // Variable initializations:
  E = 1.0;
  c = 0.3;
  dE = 0.1;
  K = 0.1;
  dE_ = 0.25;
  K_ = 05.0;
end

model bonhoeffer_1990__kinetic_parameters(b, qa, ql)

  // Variable initializations:
  b = 0.001;
  qa = 0.001;
  ql = 0.001;
end

  kinetic_parameters: bonhoeffer_1990__kinetic_parameters(b, qa, ql);

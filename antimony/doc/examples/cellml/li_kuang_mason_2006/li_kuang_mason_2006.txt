model li_model_2006__G(G, Vg, time_, I)

  // Assignment Rules:
  f2_G := Ub * ( 1.0 - exp( 1.0 * (-( G ) / ( C2 * Vg ))));
  f3_G := G / ( C3 * Vg );
  f4_I := U0 + ( Um - U0 ) / ( 1.0 + exp(-( beta ) * ln( 1.0 * ( I / ( C4 * ( 1.0 / Vi + 1.0 / ( E * ti )))))));
  f5_I := Rg / ( 1.0 + exp( alpha * ( I / ( 1.0 * Vp - 1.0 * C5 ))));

  // Rate Rules:
  G' = Gin + 1.0 * f5_I + -(( 1.0 * f2_G + f3_G * f4_I ));

  // Variable initializations:
  G = 1000.0;
  Vg = 10.0;
  Gin = 0.54;
  C2 = 144.0;
  C3 = 1000.0;
  C4 = 80.0;
  C5 = 26.0;
  U0 = 40.0;
  Um = 940.0;
  Ub = 72.0;
  beta = 1.77;
  Rg = 180.0;
  alpha = 0.29;
  ti = 100;
  E = 0.2;
  Vp = 3.0;
  Vi = 11.0;
end

model li_model_2006__I(I, time_, Vg, G)

  // Assignment Rules:
  f1_G := Rm / ( 1.0 + exp(( C1 - G ) / ( 1.0 * Vg ) / a1 ));

  // Rate Rules:
  I' = 1.0 * f1_G - di * I;

  // Variable initializations:
  I = 9000.0;
  Rm = 210.0;
  C1 = 2000.0;
  a1 = 300.0;
  di = 0.06;
end

  Cdc13_Cdc2' = k1 * mass + kcdc25 * Cdc13_P_Cdc2 + Rum1_Cdc13_Cdc2 * (kir + k4) - (kwee * Cdc13_Cdc2 + ki * Cdc13_Cdc2 * Rum1 + k2 * Cdc13_Cdc2);

  // Variable initializations:
  Cdc13_Cdc2 = 1;
  k1 = 0.03;
end

  // Rate Rules:
  Cdc13_P_Cdc2' = kwee * Cdc13_Cdc2 - (kcdc25 * Cdc13_P_Cdc2 + k2 * Cdc13_P_Cdc2);

  // Variable initializations:
  Cdc13_P_Cdc2 = 0;
end

model novak_csikasznagy_gyorffy_chen_tyson_1998_version01__Cdc25P(Cdc25P, ks, MPF, time_)

  // Assignment Rules:
  k25r := k25r_ + ks;

  // Rate Rules:
  Cdc25P' = k25 * MPF * (1 - Cdc25P) / (J25 + 1 - Cdc25P) - k25r * Cdc25P / (J25r + Cdc25P);

  // Variable initializations:
  Cdc25P = 0;
  k25 = 0.5;
  k25r_ = 0.2;
  J25 = 0.2;
  J25r = 0.2;
end

model novak_csikasznagy_gyorffy_chen_tyson_1998_version01__Wee1(Wee1, ks, MPF, time_)

  // Assignment Rules:
  kwr := kwr_ + ks;

  // Rate Rules:
  Wee1' = kwr * (1 - Wee1) / (Jwr + 1 - Wee1) - kw * MPF * Wee1 / (Jw + Wee1);

  // Variable initializations:
  Wee1 = 0;
  kw = 0.5;
  kwr_ = 0.2;
  Jw = 0.2;
  Jwr = 0.2;
end

model novak_csikasznagy_gyorffy_chen_tyson_1998_version01__Mik1(Mik1, ks, time_)

  // Assignment Rules:
  kmr := kmr_ + ks;

  // Rate Rules:
  Mik1' = kmr * (1 - Mik1) / (Jmr + 1 - Mik1) - km * Mik1 / (Jm + Mik1);

  // Variable initializations:
  Mik1 = 0;
  km = 0.1;
  kmr_ = 0;
  Jm = 0.2;
  Jmr = 0.2;
end

end

model novak_csikasznagy_gyorffy_chen_tyson_1998_version01__AAE(AAE, AAE_total, kad, MPF, time_)

  // Assignment Rules:
  kai := kai_ + kx;

  // Rate Rules:
  AAE' = (kaa + kaa_ * MPF) * (AAE_total - AAE) / (Jaa + AAE_total - AAE) - (kai * AAE / (Jai + AAE) + kad * AAE);

  // Variable initializations:
  AAE = 2;
  kaa = 0.001;
  kaa_ = 1;
  kai_ = 0.25;
  kx = 0;
  Jaa = 0.1;
  Jai = 0.1;
end

model novak_csikasznagy_gyorffy_chen_tyson_1998_version01__APC(APC, Puc1, MPF, AAE, mass, time_)

  // Rate Rules:
  APC' = (kapr + kapr_ * AAE) * (1 - APC) / (Japr + 1 - APC) - kap * (Puc1 * mass + MPF) * APC / (Jap + APC);

  // Variable initializations:
  APC = 0;
  kapr = 0.04;
  kapr_ = 3;
  kap = 4;
  Japr = 0.01;
  Jap = 0.01;
end

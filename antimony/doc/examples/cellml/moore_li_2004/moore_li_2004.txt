end

model moore_2004__Tn(Tn, kn, eta, C, time_)

  // Rate Rules:
  Tn' = sn - ( dn * Tn + kn * Tn * ( C / ( C + eta )));

  // Variable initializations:
  Tn = 1510.0;
  sn = 0.37;
  dn = 0.23;
end

model moore_2004__Te(Te, kn, eta, C, Tn, time_)

  // Rate Rules:
  Te' = alpha_n * kn * Tn * ( C / ( C + eta )) + alpha_e * Te * ( C / ( C + eta )) - ( de * Te + gamma_e * C * Te );

  // Variable initializations:
  Te = 20.0;
  alpha_n = 0.14;
  alpha_e = 0.98;
  de = 0.30;
  gamma_e = 0.057;
end

model moore_2004__C(C, Te, time_)

  // Rate Rules:
  C' = rc * C * ln( Cmax / C ) - ( dc * C + gamma_c * C * Te );

  // Variable initializations:
  C = 10000.0;
  Cmax = 230000;
  rc = 0.0057;
  dc = 0.024;
  gamma_c = 0.0034;
end

model moore_2004__model_parameters(kn, eta)

  // Variable initializations:
  kn = 0.062;
  eta = 720.0;
end

model moore_2004____main()

  // Sub-modules, and any changes to those submodules:
  environment: moore_2004__environment(time_);
  Tn: moore_2004__Tn(Tn0, kn, eta, C0, time_);
  Te: moore_2004__Te(Te0, kn, eta, C0, Tn0, time_);
  C: moore_2004__C(C0, Te0, time_);
  model_parameters: moore_2004__model_parameters(kn, eta);
end

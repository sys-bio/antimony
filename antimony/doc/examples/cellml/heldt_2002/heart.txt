model heldt_heart_2002__Environment(time_, HR, hrf)

  // Assignment Rules:
  hrf := HR / 60;

  // Variable initializations:
  HR = 70;
end

model heldt_heart_2002__LVTiming(realtime, HR, beattime, hrf)

  // Assignment Rules:
  beattime := realtime - floor(realtime / hrf) * hrf - PRint;

  // Variable initializations:
  PRint = 0.00012;
end

model heldt_heart_2002__LVElastanceFunction(time_, hrf, E_LV)

  // Assignment Rules:
  Ts := TsK * root(1 * hrf);
  E_LV := piecewise( Edia + (Esys - Edia) * (1 - cos(pi * time_ / Ts)) / 2 , ( time_ >= 0) && (time_ <= Ts ), Edia + (Esys - Edia) * (1 + cos(2 * pi * (time_ - Ts) / Ts)) / 2 , ( time_ < 1.5 * Ts) && (time_ >= Ts ), Edia );

  // Variable initializations:
  Esys = 5.6;
  Edia = 0.19;
  TsK = 0.35;
end

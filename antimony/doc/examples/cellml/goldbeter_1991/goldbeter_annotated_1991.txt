model goldbeter_1991__environment(time_)
end

model goldbeter_1991__C(C, X, kd, Kd, vi, vd, time_)

  // Rate Rules:
  C' = vi - vd * X * C / (Kd + C) - kd * C;

  // Variable initializations:
  C = 0.01;
end

model goldbeter_1991__M(M, M_star, V1, V2, K1, K2, time_)

  // Rate Rules:
  M' = V1 * M_star / (K1 + M_star) - V2 * M / (K2 + M);

  // Variable initializations:
  M = 0.01;
end

model goldbeter_1991__M_star(M_star, M)

  // Assignment Rules:
  M_star := 1 - M;
end

model goldbeter_1991__X(X, X_star, V3, V4, K3, K4, time_)

  // Rate Rules:
  X' = V3 * X_star / (K3 + X_star) - V4 * X / (K4 + X);

  // Variable initializations:
  X = 0.01;
end

model goldbeter_1991__X_star(X_star, X)

  // Assignment Rules:
  X_star := 1 - X;
end

model goldbeter_1991__model_parameters(vi, vd, kd, Kd, Kc, K1, K2, K3, K4, V1, V2, V3, V4, VM1, VM3, C, M)

  // Assignment Rules:
  V1 := VM1 * C / (Kc + C);
  V3 := M * VM3;

  // Variable initializations:
  vi = 0.025;
  vd = 0.25;
  kd = 0.01;
  Kd = 0.02;
  Kc = 0.5;
  K1 = 0.005;
  K2 = 0.005;
  K3 = 0.005;
  K4 = 0.005;
  V2 = 1.5;
  V4 = 0.5;
  VM1 = 3;
  VM3 = 1;
end

  // Sub-modules, and any changes to those submodules:
  environment: goldbeter_1991__environment(time_);

  X_star: goldbeter_1991__X_star(X_star0, X0);

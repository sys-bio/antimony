  C_int_H := 1e3 * power(10 , -(pH_int));

  // Rate Rules:
  pH_int' = 5;

  // Variable initializations:
  C_ext_Na = 1;
  C_ext_H = 5.4954e-5;
  C_ext_NH4 = 0;

  pH_int = 4;

  // Variable initializations:
  XTxP0_NHE3_Na = 1.6e-3;

  XTxP0_NHE3_NH4 = 1.6e-3;
  K_NHE3_Na = 30;

  K_NHE3_NH4 = 0.027e3;

model Proximal_Tubule_NHE3__NHE3(C_ext_Na, C_ext_H, C_ext_NH4, C_int_Na, C_int_H, C_int_NH4, XTxP0_NHE3_Na, XTxP0_NHE3_H, XTxP0_NHE3_NH4, K_NHE3_Na, K_NHE3_H, K_NHE3_NH4, J_NHE3_Na, J_NHE3_H, J_NHE3_NH4, J_NHE3_Na2, J_NHE3_H2, J_NHE3_NH42)

  // Assignment Rules:
  XTxP_NHE3_Na := XTxP0_NHE3_Na * 2 * C_int_H / (C_int_H + 1e-3 );
  XTxP_NHE3_H := XTxP0_NHE3_H * 2 * C_int_H / (C_int_H + 1e-3 );
  XTxP_NHE3_NH4 := XTxP0_NHE3_NH4 * 2 * C_int_H / (C_int_H + 1e-3 );
  alpha_ext_Na := C_ext_Na / K_NHE3_Na;
  beta_ext_H := C_ext_H / K_NHE3_H;
  gamma_ext_NH4 := C_ext_NH4 / K_NHE3_NH4;
  alpha_int_Na := C_int_Na / K_NHE3_Na;
  beta_int_H := C_int_H / K_NHE3_H;
  gamma_int_NH4 := C_int_NH4 / K_NHE3_NH4;
  sum_NHE3 := (1 + alpha_ext_Na + beta_ext_H + gamma_ext_NH4) * (XTxP_NHE3_Na * alpha_int_Na + XTxP_NHE3_H * beta_int_H + XTxP_NHE3_NH4 * gamma_int_NH4) + (1 + alpha_int_Na + beta_int_H + gamma_int_NH4) * (XTxP_NHE3_Na * alpha_ext_Na + XTxP_NHE3_H * beta_ext_H + XTxP_NHE3_NH4 * gamma_ext_NH4);
  J_NHE3_Na := XTxP_NHE3_Na * XTxP_NHE3_H / sum_NHE3 * (alpha_ext_Na * beta_int_H - alpha_int_Na * beta_ext_H) + XTxP_NHE3_Na * XTxP_NHE3_NH4 / sum_NHE3 * (alpha_ext_Na * gamma_int_NH4 - alpha_int_Na * gamma_ext_NH4);
  J_NHE3_H := XTxP_NHE3_Na * XTxP_NHE3_H / sum_NHE3 * (alpha_int_Na * beta_ext_H - alpha_ext_Na * beta_int_H) + XTxP_NHE3_H * XTxP_NHE3_NH4 / sum_NHE3 * (beta_ext_H * gamma_int_NH4 - beta_int_H * gamma_ext_NH4);
  J_NHE3_NH4 := XTxP_NHE3_Na * XTxP_NHE3_NH4 / sum_NHE3 * (alpha_int_Na * gamma_ext_NH4 - alpha_ext_Na * gamma_int_NH4) + XTxP_NHE3_H * XTxP_NHE3_NH4 / sum_NHE3 * (gamma_ext_NH4 * beta_int_H - beta_ext_H * gamma_int_NH4);
  J_NHE3_Na_Max := XTxP0_NHE3_Na * XTxP0_NHE3_H / (XTxP0_NHE3_Na + XTxP0_NHE3_H);
  plot := J_NHE3_Na / J_NHE3_Na_Max;
  J_NHE3_Na2 := XTxP0_NHE3_Na * XTxP0_NHE3_H / sum0_NHE3 * (alpha_ext_Na * beta_int_H - alpha_int_Na * beta_ext_H) + XTxP0_NHE3_Na * XTxP0_NHE3_NH4 / sum_NHE3 * (alpha_ext_Na * gamma_int_NH4 - alpha_int_Na * gamma_ext_NH4);
  J_NHE3_H2 := XTxP0_NHE3_Na * XTxP0_NHE3_H / sum0_NHE3 * (alpha_int_Na * beta_ext_H - alpha_ext_Na * beta_int_H) + XTxP0_NHE3_H * XTxP0_NHE3_NH4 / sum_NHE3 * (beta_ext_H * gamma_int_NH4 - beta_int_H * gamma_ext_NH4);
  J_NHE3_NH42 := XTxP0_NHE3_Na * XTxP0_NHE3_NH4 / sum0_NHE3 * (alpha_int_Na * gamma_ext_NH4 - alpha_ext_Na * gamma_int_NH4) + XTxP0_NHE3_H * XTxP0_NHE3_NH4 / sum_NHE3 * (gamma_ext_NH4 * beta_int_H - beta_ext_H * gamma_int_NH4);
  sum0_NHE3 := (1 + alpha_ext_Na + beta_ext_H + gamma_ext_NH4) * (XTxP0_NHE3_Na * alpha_int_Na + XTxP0_NHE3_H * beta_int_H + XTxP0_NHE3_NH4 * gamma_int_NH4) + (1 + alpha_int_Na + beta_int_H + gamma_int_NH4) * (XTxP0_NHE3_Na * alpha_ext_Na + XTxP0_NHE3_H * beta_ext_H + XTxP0_NHE3_NH4 * gamma_ext_NH4);
  plot2 := J_NHE3_Na2 / J_NHE3_Na_Max;
end

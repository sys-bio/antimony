model lockwood_2006__environment(time_)
end

model lockwood_2006__population_pharmacodynamics_model(time_, ADAS_Cog_p, PD_CeA)

  // Assignment Rules:
  S := S0 + alpha * time_ + ADAS_Cog_p + PD_CeA + epsilon;

  // Variable initializations:
  S0 = 30;
  alpha = 0.0164;
  epsilon = 0.0;
end

model lockwood_2006__placebo_response_model(ADAS_Cog_p, time_)

  // Assignment Rules:
  ADAS_Cog_p := beta_P * Keq_p / (Keq_p - Kel_p) * (exp(-(Kel_p) * time_) - exp(-(Keq_p) * time_));
  Keq_p := ln(2 ) / t_half_eq_p;
  Kel_p := ln(2 ) / t_half_el_p;

  // Variable initializations:
  beta_P = -3;
  t_half_el_p = 7;
  t_half_eq_p = 6;
end

model lockwood_2006__pharmacokinetic_model(CL)

  // Assignment Rules:
  CL := 2268.0 * exp(-(0.0135 ) * (age - 40 )) * smk;

  // Variable initializations:
  smk = 1;
  age = 40;
end

model lockwood_2006__drop_out_model(time_)

  // Assignment Rules:
  Sv := exp(-(0.00145 ) * time_);
end

model lockwood_2006__drug_response_model(PD_CeA)

  // Assignment Rules:
  PD_CeA := beta_a * CeA;

  // Variable initializations:
  beta_a = -0.047;
  CeA = 25;
end

model lockwood_2006__drug_clearance(time_, CL)

  // Rate Rules:
  CC' = (k_ab * A_in - (CL * CC + CL_ic * (CC - PC))) / Vc;
  PC' = CL_ic * (CC - PC) / Vp;
  A_in' = -(115.44 ) * A_in;

  // Variable initializations:
  CC = 0;
  PC = 0;
  Vc = 172;
  Vp = 222;
  A_in = 25;
  k_ab = 115.44;
  CL_ic = 763.2;
end

model lockwood_2006____main()

  // Sub-modules, and any changes to those submodules:
  environment: lockwood_2006__environment(time_);
  population_pharmacodynamics_model: lockwood_2006__population_pharmacodynamics_model(time_, ADAS_Cog_p, PD_CeA);
  placebo_response_model: lockwood_2006__placebo_response_model(ADAS_Cog_p, time_);
  pharmacokinetic_model: lockwood_2006__pharmacokinetic_model(CL);
  drop_out_model: lockwood_2006__drop_out_model(time_);
  drug_response_model: lockwood_2006__drug_response_model(PD_CeA);
  drug_clearance: lockwood_2006__drug_clearance(time_, CL);
end

model borghans_model_1997c__V_in(V_in, beta)

  // Assignment Rules:
  V_in := v_0 + v_1 * beta;

  // Variable initializations:
  v_0 = 0.015;
  v_1 = 0.012;
end

model borghans_model_1997c__V_2i(Z, V_2i)

  // Assignment Rules:
  V_2i := V_M2i * (power(Z, 2 ) / (power(K_2i, 2 ) + power(Z, 2 )));

  // Variable initializations:
  V_M2i = 3.1;
  K_2i = 0.005;
end

model borghans_model_1997c__V_3i(Z, Y, V_3i)

  // Assignment Rules:
  V_3i := V_M3i * (power(Y, 2 ) / (power(K_3y, 2 ) + power(Y, 2 ))) * (power(Z, 2 ) / (power(K_3z, 2 ) + power(Z, 2 )));

  // Variable initializations:
  V_M3i = 25;
  K_3z = 0.022;
  K_3y = 0.065;
end

model borghans_model_1997c__V_2s(Z, V_2s)

  // Assignment Rules:
  V_2s := V_M2s * (power(Z, 2 ) / (power(K_2s, 2 ) + power(Z, 2 )));

  // Variable initializations:
  V_M2s = 1.5;
  K_2s = 0.0265;
end

model borghans_model_1997c__V_3s(X, V_3s, beta)

  // Assignment Rules:
  V_3s := beta * V_M3s * (power(X, 2 ) / (power(K_3s, 2 ) + power(X, 2 )));

  // Variable initializations:
  V_M3s = 0.169;
  K_3s = 0.1;
end

model borghans_model_1997c__Ca_flux(time_, beta, Z, Y, X, V_in, V_2i, V_3i, V_2s, V_3s)

  // Sub-modules, and any changes to those submodules:
  V_in_mod: borghans_model_1997c__V_in(V_in, beta);

  V_2s_mod: borghans_model_1997c__V_2s(Z, V_2s);
  V_3s_mod: borghans_model_1997c__V_3s(X, V_3s, beta);

  // Variable initializations:

end

model borghans_model_1997c__Ca(time_, Z, Y, X, V_in, V_2i, V_3i, V_2s, V_3s)

  // Sub-modules, and any changes to those submodules:
  Ca_flux: borghans_model_1997c__Ca_flux(time_, beta, Z, Y, X, V_in, V_2i, V_3i, V_2s, V_3s);

  // Rate Rules:
  Z' = V_in + -(V_2i) + V_3i + K_f * Y + -(V_2s) + V_3s + K_f * X + -((K * Z));
  Y' = V_2i + -(V_3i) + -((K_f * Y));
  X' = V_2s + -(V_3s) + -((K_f * X));

  // Variable initializations:
  Z = 0.0;
  Y = 0.0;
  X = 0.5;
  K_f = 0.5;
  K = 1;
end

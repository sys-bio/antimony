end

model halloy_2002__L(L, mu_T, mu_L, mu_A, epsilon, M, T, time_)

  // Assignment Rules:
  L_0 := mu_L / (mu_A + mu_T + mu_L);

  // Rate Rules:
  L' = 1 / mu_T * T - 1 / mu_L * L + epsilon * M;

  // Variable initializations:
  L = 0.33;
end

model halloy_2002__A(A, mu_A, mu_L, mu_T, L, time_)

  // Assignment Rules:
  A_0 := mu_A / (mu_A + mu_T + mu_L);

  // Rate Rules:
  A' = 1 / mu_L * L - 1 / mu_A * A;

  // Variable initializations:
  A = 0.34;
end

model halloy_2002__T(T, mu_A, mu_T, mu_L, A, time_)

  // Assignment Rules:
  T_0 := mu_T / (mu_A + mu_T + mu_L);

  // Rate Rules:
  T' = 1 / mu_A * A - 1 / mu_T * T;

  // Variable initializations:
  T = 0.33;
end

model halloy_2002__M(M, epsilon, T, time_)

  // Rate Rules:
  M' = epsilon * T;

  // Variable initializations:
  M = 0;
end

model halloy_2002__reaction_constants(mu_L, mu_A, mu_T, epsilon)

  // Variable initializations:
  mu_L = 2;
  mu_A = 3.6;
  mu_T = 2.4;
  epsilon = 0;
end

model halloy_2002____main()

  // Sub-modules, and any changes to those submodules:
  environment: halloy_2002__environment(time_);
  L: halloy_2002__L(L0, mu_T, mu_L, mu_A, epsilon, M0, T0, time_);
  A: halloy_2002__A(A0, mu_A, mu_L, mu_T, L0, time_);
  T: halloy_2002__T(T0, mu_A, mu_T, mu_L, A0, time_);
  M: halloy_2002__M(M0, epsilon, T0, time_);
  reaction_constants: halloy_2002__reaction_constants(mu_L, mu_A, mu_T, epsilon);
end

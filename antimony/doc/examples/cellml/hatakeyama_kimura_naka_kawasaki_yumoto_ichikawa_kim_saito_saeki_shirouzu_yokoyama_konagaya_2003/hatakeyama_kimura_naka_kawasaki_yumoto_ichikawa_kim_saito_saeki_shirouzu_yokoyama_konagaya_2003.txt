  GS' = -((kf7 * R_ShP - kb7 * R_ShGS)) + (kf9 * ShGS - kb9 * GS * ShP);

  R_PI3Kstar' = kf24 * R_PI3K - kb24 * R_PI3Kstar - (kf25 * R_PI3Kstar - kb25 * RP * PI3Kstar);

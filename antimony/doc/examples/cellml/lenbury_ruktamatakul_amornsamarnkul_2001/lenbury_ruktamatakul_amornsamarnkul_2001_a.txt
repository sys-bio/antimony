model lenbury_2001__x(x, z, y, time_)

  // Rate Rules:
  x' = z * (r_1 * y + -(r_2) * x + c_1);

  // Variable initializations:
  x = 4.0;
  r_1 = 0.2;
  r_2 = 0.1;
  c_1 = 0.1;
end

model lenbury_2001__y(y, epsilon, z, x, time_)

  // Rate Rules:
  y' = epsilon * (r_3 / z - r_4 * x + c_2);

  // Variable initializations:
  y = 0.0;
  r_3 = 0.1;
  r_4 = 0.1;
  c_2 = 0.1;
end

model lenbury_2001__z(z, y, time_)

  // Rate Rules:
  z' = r_5 * (y - y_hat) * (z_hat - z) + r_6 * z * (z_hat - z) - r_7 * z;

  // Variable initializations:
  z = 1.0;
  r_5 = 0.1;
  r_6 = 0.1;
  r_7 = 0.05;
  z_hat = 2.0;
  y_hat = 1.24;
end

model lenbury_2001__model_constants(epsilon)

  // Variable initializations:
  epsilon = 0.1;
end

model lenbury_2001____main()

  // Sub-modules, and any changes to those submodules:
  environment: lenbury_2001__environment(time_);
  x: lenbury_2001__x(x0, z0, y0, time_);
  y: lenbury_2001__y(y0, epsilon, z0, x0, time_);
  z: lenbury_2001__z(z0, y0, time_);
  model_constants: lenbury_2001__model_constants(epsilon);
end

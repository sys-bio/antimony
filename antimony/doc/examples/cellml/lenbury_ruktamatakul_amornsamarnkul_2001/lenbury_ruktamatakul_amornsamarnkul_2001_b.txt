model lenbury_ruktamatakul_amornsamarnkul_model_2001_version01__x(x, z, y, time_)

  // Rate Rules:
  x' = z * ( r1 * y + -(( r2 * x )) + c1 );

  // Variable initializations:
  x = 4.4;
  r1 = 0.2;
  r2 = 0.1;
  c1 = 0.1;
end

model lenbury_ruktamatakul_amornsamarnkul_model_2001_version01__y(y, epsilon, u, z, x, time_)

  // Rate Rules:
  y' = epsilon * ( r3 / z + -(( r4 * x )) + c2 + c3 * u );

  // Variable initializations:
  y = 1.25;
  r3 = 0.1;
  r4 = 0.1;
  c2 = 0.1;
  c3 = 0.005;
end

model lenbury_ruktamatakul_amornsamarnkul_model_2001_version01__z(z, y, epsilon, time_)

  // Assignment Rules:
  z := piecewise( 1e-7 , abs( zReal ) < 1e-7 , zReal );

  // Rate Rules:
  zReal' = epsilon * delta * ( r5 * ( y - y_ ) * ( z_ - z ) + r6 * z * ( z_ - z ) - r7 * z );

  // Variable initializations:
  r5 = 0.1;
  r6 = 0.1;
  r7 = 0.05;
  z_ = 2.0;
  y_ = 1.24;
  delta = 0.01;
end

model lenbury_ruktamatakul_amornsamarnkul_model_2001_version01__u(u, time_)

  // Rate Rules:
  u' = -(( omega * v ));
  v' = omega * u;

  // Variable initializations:
  omega = 0.05;
end

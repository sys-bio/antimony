model lenbury_2001__environment(time_)
end

model lenbury_2001__x(x, z, y, time_)

  // Rate Rules:
  x' = z * (r1 * y + -(r2) * x + c1);

  // Variable initializations:
  x = 5;
  r1 = 0.15;
  r2 = 0.12;
  c1 = 0.1;
end

model lenbury_2001__y(y, epsilon, u, z, x, time_)

  // Rate Rules:
  y' = epsilon * (r3 / z + -(r4) * x + c2 + c3 * u);

  // Variable initializations:
  y = 0;
  r3 = 0.05;
  r4 = 0.03;
  c2 = 0.1;
  c3 = 0.005;
end

model lenbury_2001__z(z, y, epsilon, time_)

  // Rate Rules:
  z' = epsilon * delta * (r5 * (y - y_) * (z_ - z) + r6 * z * (z_ - z) - r7 * z);

  // Variable initializations:
  z = 1;
  r5 = 0.09;
  r6 = 0.1;
  r7 = 0.05;
  z_ = 1.01;
  y_ = 1.08;
  delta = 0.01;
end

model lenbury_2001__u(u, time_)

  // Rate Rules:
  u' = -(omega) * v;
  v' = omega * u;

  // Variable initializations:
  u = 1;
  omega = 0.05;
  v = 0;
end

model lenbury_2001__model_constants(epsilon)

  // Variable initializations:
  epsilon = 0.1;
end

model lenbury_2001____main()

  // Sub-modules, and any changes to those submodules:
  environment: lenbury_2001__environment(time_);
  x: lenbury_2001__x(x0, z0, y0, time_);
  y: lenbury_2001__y(y0, epsilon, u0, z0, x0, time_);
  z: lenbury_2001__z(z0, y0, epsilon, time_);
  u: lenbury_2001__u(u0, time_);
  model_constants: lenbury_2001__model_constants(epsilon);
end

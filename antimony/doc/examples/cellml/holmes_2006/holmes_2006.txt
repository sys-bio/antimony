model holmes_2006__contraction(time_)

  // Assignment Rules:
  L := piecewise( 1 , time_ <= 1 , 0.92 , ( time_ > 1) && (time_ < 5 ), 0.9 );
  v := -(b) * (Po - P) / (P + a);
  L_se := L - L_ce;
  P := alpha * (L_se - L_se_o);

  // Rate Rules:
  L_ce' = v;

  // Variable initializations:
  a = 37.24;
  b = 0.325;
  Po = 144.9;
  alpha = 1449.027;
  L_se_o = 0.3;
  L_ce = 0.7;
end

  environment: holmes_2006__environment(time_);

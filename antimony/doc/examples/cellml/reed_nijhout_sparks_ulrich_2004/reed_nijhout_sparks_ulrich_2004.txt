model reed_2004__environment(time_, Metin)

  // Assignment Rules:
  Metin := piecewise( 200 , ( time_ < 2) || (time_ >= 8 ), 300 , ( time_ >= 2) && (time_ < 5 ), 100 , ( time_ >= 5) && (time_ < 8 ), 200 );
end

model reed_2004__Met(Met, V_MS, V_BHMT, V_MATI, V_MATIII, Metin, time_)

  // Rate Rules:
  Met' = V_MS + V_BHMT + Metin - (V_MATI + V_MATIII);

  // Variable initializations:
  Met = 53.5;
end

model reed_2004__AdoMet(AdoMet, V_METH, V_GNMT, V_MATI, V_MATIII, time_)

  // Rate Rules:
  AdoMet' = V_MATI + V_MATIII - (V_METH + V_GNMT);

  // Variable initializations:
  AdoMet = 137.6;
end

model reed_2004__AdoHcy(AdoHcy, V_METH, V_GNMT, V_AH, time_)

  // Rate Rules:
  AdoHcy' = V_METH + V_GNMT - V_AH;

  // Variable initializations:
  AdoHcy = 13.2;
end

model reed_2004__Hcy(Hcy, V_AH, V_CBS, V_MS, V_BHMT, time_)

  // Rate Rules:
  Hcy' = V_AH - (V_CBS + V_MS + V_BHMT);

  // Variable initializations:
  Hcy = 0.88;
end

model reed_2004__V_MATI(V_MATI, Met, AdoMet)

  // Assignment Rules:
  V_MATI := V_MATImax / (1 + Km_MATI / Met * (1 + AdoMet / Ki_MATI));

  // Variable initializations:
  V_MATImax = 561;
  Km_MATI = 41;
  Ki_MATI = 50;
end

model reed_2004__V_MATIII(V_MATIII, AdoMet, Met)

  // Assignment Rules:
  V_MATIII := V_MATIIImax / (1 + Km1_MATIII * Km2_MATIII / (power(Met, 2 ) + Met * Km2_MATIII));
  Km1_MATIII := 20000 / (1 + 5.7 * power(AdoMet / (AdoMet + 600 ), 2 ));

  // Variable initializations:
  V_MATIIImax = 22870;
  Km2_MATIII = 21.1;
end

model reed_2004__V_GNMT(V_GNMT, AdoMet, AdoHcy)

  // Assignment Rules:
  V_GNMT := V_GNMTmax / (1 + power(Km_GNMT / AdoMet, 2.3 )) * 1 / (1 + AdoHcy / Ki_GNMT);

  // Variable initializations:
  V_GNMTmax = 10600;
  Km_GNMT = 4500;
  Ki_GNMT = 20;
end

model reed_2004__V_METH(V_METH, AdoMet, AdoHcy)

  // Assignment Rules:
  V_METH := V_METHmax / (1 + Km1_METH / AdoMet + Km2_METH_A + Km2_METH_A * Km1_METH / AdoMet);
  Km1_METH := 1 * (1 + AdoHcy / 4 );

  // Variable initializations:
  V_METHmax = 4521;
  Km2_METH_A = 10;
end

model reed_2004__V_MS(V_MS, Hcy)

  // Assignment Rules:
  V_MS := V_MSmax * five_mTHF * Hcy / (Kd_MS * Km_Hcy_MS + Km_Hcy_MS * five_mTHF + Km_five_mTHF_MS * Hcy + five_mTHF * Hcy);

  // Variable initializations:
  five_mTHF = 5.2;
  V_MSmax = 500;
  Kd_MS = 1;
  Km_Hcy_MS = 0.1;
  Km_five_mTHF_MS = 25;
end

model reed_2004__V_AH(V_AH, AdoHcy, Hcy)

  // Assignment Rules:
  V_AH := alpha1 * (AdoHcy - alpha2 * Hcy);

  // Variable initializations:
  alpha1 = 100;
  alpha2 = 10;
end

model reed_2004__V_CBS(V_CBS, AdoHcy, Hcy, AdoMet)

  // Assignment Rules:
  V_CBS := (beta1 * (AdoMet + AdoHcy) - beta2) * Hcy;

  // Variable initializations:
  beta1 = 1.7;
  beta2 = 30;
end

model reed_2004__V_BHMT(V_BHMT, AdoMet, AdoHcy, Hcy)

  // Assignment Rules:
  V_BHMT := (0.7 - 0.025 * (AdoMet + AdoHcy - 150 )) * V_BHMTmax * Hcy / (Km_BHMT + Hcy);

  // Variable initializations:
  V_BHMTmax = 2500;
  Km_BHMT = 12;
end

model reed_2004____main()

  // Sub-modules, and any changes to those submodules:
  environment: reed_2004__environment(time_, Metin);
  Met: reed_2004__Met(Met0, V_MS0, V_BHMT0, V_MATI0, V_MATIII0, Metin, time_);
  AdoMet: reed_2004__AdoMet(AdoMet0, V_METH0, V_GNMT0, V_MATI0, V_MATIII0, time_);
  AdoHcy: reed_2004__AdoHcy(AdoHcy0, V_METH0, V_GNMT0, V_AH0, time_);
  Hcy: reed_2004__Hcy(Hcy0, V_AH0, V_CBS0, V_MS0, V_BHMT0, time_);
  V_MATI: reed_2004__V_MATI(V_MATI0, Met0, AdoMet0);
  V_MATIII: reed_2004__V_MATIII(V_MATIII0, AdoMet0, Met0);
  V_GNMT: reed_2004__V_GNMT(V_GNMT0, AdoMet0, AdoHcy0);
  V_METH: reed_2004__V_METH(V_METH0, AdoMet0, AdoHcy0);
  V_MS: reed_2004__V_MS(V_MS0, Hcy0);
  V_AH: reed_2004__V_AH(V_AH0, AdoHcy0, Hcy0);
  V_CBS: reed_2004__V_CBS(V_CBS0, AdoHcy0, Hcy0, AdoMet0);
  V_BHMT: reed_2004__V_BHMT(V_BHMT0, AdoMet0, AdoHcy0, Hcy0);
end

model oomens_maenhout_vanoijen_drost_baaijens_2003_version01__r(r, c, mu)

  // Assignment Rules:
  r := power( c , 2.0 ) / (power( c , 2.0 ) + mu * c + power( mu , 2.0 ));
end

model oomens_maenhout_vanoijen_drost_baaijens_2003_version01__Ct(Ct, c_, rho, c, r, mu, chi, time_)

  // Assignment Rules:
  Ct_initial := c + 2.0 * b * Q0 + r * ( 2.0 + mu / c ) * ( 1.0 - b * Q0 );

  // Rate Rules:
  Ct' = rho * ( 1.0 - c / c_ ) * chi - power( tau0 , -1.0 ) * ( c / ( c + km ));

  // Variable initializations:
  km = 0.006;
  tau0 = 9.2;
end

model oomens_maenhout_vanoijen_drost_baaijens_2003_version01__Ca_in(Ca_in, rho, chi, c_, c, time_)

  // Rate Rules:
  Ca_in' = rho * ( 2.0 - c / c_ ) * chi;
end

model oomens_maenhout_vanoijen_drost_baaijens_2003_version01__chi(chi, time_)

  // Assignment Rules:
  chi := sum(domainofapplication( i ), chi_ * ( time_ - time_i ));
  chi_ := (exp(-( time_ ) / tau1 ) - exp(-( time_ ) / tau2 )) * ( tau1 - tau2 );

  // Variable initializations:
  tau1 = 0.005;
  tau2 = 0.001;
end

model oomens_maenhout_vanoijen_drost_baaijens_2003_version01__model_parameters(mu, c, c_, rho)

  // Variable initializations:
  mu = 0.2;
  rho = 2.9;
end

model oomens_maenhout_vanoijen_drost_baaijens_2003_version01____main()

  // Sub-modules, and any changes to those submodules:
  environment: oomens_maenhout_vanoijen_drost_baaijens_2003_version01__environment(time_);
  r: oomens_maenhout_vanoijen_drost_baaijens_2003_version01__r(r0, c, mu);
  Ct: oomens_maenhout_vanoijen_drost_baaijens_2003_version01__Ct(Ct0, c_, rho, c, r0, mu, chi0, time_);
  Ca_in: oomens_maenhout_vanoijen_drost_baaijens_2003_version01__Ca_in(Ca_in0, rho, chi0, c_, c, time_);
  chi: oomens_maenhout_vanoijen_drost_baaijens_2003_version01__chi(chi0, time_);
  model_parameters: oomens_maenhout_vanoijen_drost_baaijens_2003_version01__model_parameters(mu, c, c_, rho);
end

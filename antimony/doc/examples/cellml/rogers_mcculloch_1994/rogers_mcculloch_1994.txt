model rogers_mcculloch_1994_version01__membrane_potential(t, Cm, Vr, Vth, Vp, Istim, Iion, Vm, u)

  // Assignment Rules:
  u := (Vm - Vr) / (Vp - Vr);

  // Rate Rules:
  Vm' = (Istim - Iion) / Cm;

  // Variable initializations:
  Vm = -85;
end

model rogers_mcculloch_1994_version01__ionic_current(Vr, Vth, Vp, c1, c2, v, u, Iion)

  // Assignment Rules:
  Iion := c1 * u * (u - (Vth - Vr) / (Vp - Vr)) * (u - 1.0 ) + c2 * u * v;
end

model rogers_mcculloch_1994_version01__recovery_variable(t, b, d, u, v)

  // Rate Rules:
  v' = b * (u - d * v);

  // Variable initializations:
  v = 0.0;
end

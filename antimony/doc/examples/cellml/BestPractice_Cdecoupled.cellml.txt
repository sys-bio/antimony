model cellmlmod_time()

  //Other declarations:
  var t;
end

model cellmlmod_geometry()

  // Variable initializations:
  R = 5;

  //Other declarations:
  var R;
end

model cellmlmod_C_interface()

  // Assignment Rules:
  Cpc := R / 602.2
  JTotal := JPlusRx1 - Cpc * JMinusRx2

  //Other declarations:
  var JPlusRx1, JMinusRx2, Cpc, JTotal, R;
end

model cellmlmod_C()

  // Rate Rules:
  C' = JGain

  // Variable initializations:
  C = 0;

  //Other declarations:
  var C, JGain, t;
end

model cellmlmod_Cformation()

  // Assignment Rules:
  J1 := kf1 * A * B - kr1 * C

  // Rate Rules:
  A' = -(J1)
  B' = -(J1)

  // Variable initializations:
  A = 10;
  B = 5;
  kf1 = 0.2;
  kr1 = 0.05;

  //Other declarations:
  var A, B, J1, kf1, kr1, C, t;
end

model cellmlmod_Eformation()

  // Assignment Rules:
  J2 := kf2 * C * D - kr2 * E

  // Rate Rules:
  D' = -(J2)
  E' = J2

  // Variable initializations:
  D = 4;
  E = 0;
  kf2 = 0.006;
  kr2 = 0.001;

  //Other declarations:
  var D, E, J2, kf2, kr2, C, t;
end

model BestPractice_Cdecoupled()

  // Sub-modules, and any changes to those submodules:
  time: cellmlmod_time()
  geometry: cellmlmod_geometry()
  C_interface: cellmlmod_C_interface()
  C: cellmlmod_C()
  Cformation: cellmlmod_Cformation()
  Eformation: cellmlmod_Eformation()
  time.t is C.t;
  geometry.R is C_interface.R;
  Cformation.J1 is C_interface.JPlusRx1;
  Eformation.J2 is C_interface.JMinusRx2;
  C_interface.JTotal is C.JGain;
  C.C is Cformation.C;
  Cformation.C is Eformation.C;
  C.t is Cformation.t;
  Cformation.t is Eformation.t;
end

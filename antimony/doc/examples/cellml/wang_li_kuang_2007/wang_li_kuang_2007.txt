model wang_model_2007__G(G, f2_G, f3_G, f4_I, f5_I, time_)

  // Assignment Rules:
  Gin := piecewise( 0.05 + 5.0 / 15.0 * time_ , ( time_ < 15.0) && (time_ >= 0.0 ), 0.05 + 5.0 * (( 45.0 - time_ ) / ( 45.0 - 15.0 )) , ( time_ < 45.0) && (time_ >= 15.0 ), 0.05 , ( time_ <= 240.0) && (time_ >= 45.0 ), 0.05 + 5.0 / 255.0 * time_ , ( time_ > 240.0) && (time_ < 255.0 ), 0.05 + 5.0 * (( 285.0 - time_ ) / ( 285.0 - 255.0 )) , ( time_ < 285.0) && (time_ >= 255.0 ), 0.05 , ( time_ <= 480.0) && (time_ >= 285.0 ), 0.05 + 5.0 / 495.0 * time_ , ( time_ < 495.0) && (time_ > 480.0 ), 0.05 + 5.0 * (( 525.0 - time_ ) / ( 525.0 - 495.0 )) , ( time_ < 525.0) && (time_ >= 495.0 ), 0.05 , ( time_ <= 720.0) && (time_ >= 525.0 ), 0.05 + 5.0 / 735.0 * time_ , ( time_ > 720.0) && (time_ < 735.0 ), 0.05 + 5.0 * (( 765.0 - time_ ) / ( 765.0 - 735.0 )) , ( time_ < 765.0) && (time_ >= 735.0 ), 0.05 , ( time_ <= 960.0) && (time_ >= 765.0 ) );

  // Rate Rules:
  G' = Gin + 1 * f5_I - ( 1 * f2_G + f3_G * f4_I );

  // Variable initializations:
  G = 150.0;
end

model wang_model_2007__I(I, di, time_)

  // Assignment Rules:
  Iin := piecewise( 0.25 , ( time_ <= 30.0) && (time_ >= 0.0 ), 0.25 + 1.0 * ( 1.0 + ( time_ - 120.0 ) / 90.0 ) , ( time_ < 120.0) && (time_ >= 30.0 ), 0.25 + 1.0 * ( 1.0 - 0.5 * (( time_ - 120.0 ) / 120.0 )) , ( time_ < 240.0) && (time_ >= 120.0 ), 0.25 + 0.5 * ( 1.0 - ( time_ - 240.0 ) / 240.0 ) , ( time_ <= 480.0) && (time_ >= 240.0 ), 0.25 , ( time_ < 530.0) && (time_ > 480.0 ), 0.25 + 1.0 * ( 1.0 + ( time_ - 620.0 ) / 90.0 ) , ( time_ < 620.0) && (time_ >= 530.0 ), 0.25 + 1.0 * ( 1.0 - 0.5 * (( time_ - 620.0 ) / 120.0 )) , ( time_ < 740.0) && (time_ >= 620.0 ), 0.25 + 0.5 * ( 1.0 - ( time_ - 740.0 ) / 240.0 ) , ( time_ <= 980.0) && (time_ >= 740.0 ) );

  // Rate Rules:
  I' = Iin - di * I;

  // Variable initializations:
  I = 5.0;
end

model wang_model_2007__f2_G(f2_G, G, C2, Ub, Vg)

  // Assignment Rules:
  f2_G := Ub * ( 1.0 - exp(-( G ) / ( C2 * Vg * 1.0 )));
end

model wang_model_2007__f3_G(f3_G, G, C3, Vg)

  // Assignment Rules:
  f3_G := G / ( C3 * Vg );
end

model wang_model_2007__f4_I(f4_I, I, C4, Vi, U0, Um, beta, ti)

  // Assignment Rules:
  f4_I := U0 + ( Um - U0 ) / ( 1.0 + exp(-( beta ) * ln( I / ( C4 * ( 1.0 / Vi + 1.0 / ( 0.2 * ti ))))));
end

model wang_model_2007__f5_I(f5_I, Vp, I, C5, Rg, alpha)

  // Assignment Rules:
  f5_I := Rg / ( 1.0 + exp( alpha * ( I / ( 1.0 * ( 1.0 * Vp - C5 )))));
end

model renal_Na_H_exchanger_model__E(E, k1, k2, k3, k4, k5, k6, ENa, ENH4, EH, H, Na, NH4, time_)

  // Rate Rules:
  E' = k2 * ENa + k4 * EH + k6 * ENH4 - ( k1 * Na * E + k3 * H * E + k5 * NH4 * E );
end

model renal_Na_H_exchanger_model__ENa(ENa, k1, k2, k13, k14, ENa_, E, Na, time_)

  // Rate Rules:
  ENa' = k1 * E * Na + k14 * ENa_ - ( k2 * ENa + k13 * ENa );
end

model renal_Na_H_exchanger_model__EH(EH, k3, k4, k15, k16, EH_, E, H, time_)

  // Rate Rules:
  EH' = k3 * H * E + k16 * EH_ - ( k4 * EH + k15 * EH );
end

model renal_Na_H_exchanger_model__ENH4(ENH4, k5, k6, k17, k18, ENH4_, E, NH4, time_)

  // Rate Rules:
  ENH4' = k5 * NH4 * E + k18 * ENH4_ - ( k6 * ENH4 + k17 * ENH4 );
end

model renal_Na_H_exchanger_model__E_(E_, k7, k8, k9, k10, k11, k12, ENa_, ENH4_, EH_, H_, Na_, NH4_, time_)

  // Rate Rules:
  E_' = k8 * ENa_ + k10 * EH_ + k12 * ENH4_ - ( k7 * Na_ * E_ + k9 * H_ * E_ + k11 * NH4_ * E_ );
end

model renal_Na_H_exchanger_model__ENa_(ENa_, k7, k8, k13, k14, ENa, E_, Na_, time_)

  // Rate Rules:
  ENa_' = k7 * E_ * Na_ + k13 * ENa - ( k8 * ENa_ + k14 * ENa_ );
end

model renal_Na_H_exchanger_model__EH_(EH_, k9, k10, k15, k16, EH, E_, H_, time_)

  // Rate Rules:
  EH_' = k9 * H_ * E_ + k15 * EH - ( k10 * EH_ + k16 * EH_ );
end

model renal_Na_H_exchanger_model__ENH4_(ENH4_, k11, k12, k17, k18, ENH4, E_, NH4_, time_)

  // Rate Rules:
  ENH4_' = k11 * NH4_ * E_ + k17 * ENH4 - ( k12 * ENH4_ + k18 * ENH4_ );
end

  k16 := 4.8E-1 * W;

  k18 := 2.41E0 * W;
  W := H_ / ( H_ + KH );

  k1 = 1.0E8;
  k2 = 3.27E6;
  k3 = 1.0E8;
  k4 = 3.52E0;

  k8 = 3.27E6;

model renal_anion_exchanger_model__environment(time_)
end

model renal_anion_exchanger_model__E(E, k1, k2, k3, k4, ECl, EHCO3, HCO3, Cl, time_)

  // Rate Rules:
  E' = k2 * ECl + k4 * EHCO3 - ( k1 * Cl * E + k3 * HCO3 * E );
end

model renal_anion_exchanger_model__ECl(ECl, k1, k2, k9, k10, ECl_, E, Cl, time_)

  // Rate Rules:
  ECl' = k1 * Cl * E + k10 * ECl_ - ( k2 * ECl + k9 * ECl );
end

model renal_anion_exchanger_model__EHCO3(EHCO3, k11, k12, k3, k4, E, EHCO3_, HCO3, time_)

  // Rate Rules:
  EHCO3' = k3 * HCO3 * E + k12 * EHCO3_ - ( k4 * EHCO3 + k11 * EHCO3 );
end

model renal_anion_exchanger_model__E_(E_, k5, k6, k7, k8, ECl_, EHCO3_, HCO3_, Cl_, time_)

  // Rate Rules:
  E_' = k6 * ECl_ + k8 * EHCO3_ - ( k5 * Cl_ * E_ + k7 * HCO3_ * E_ );
end

model renal_anion_exchanger_model__ECl_(ECl_, k5, k6, k9, k10, ECl, E_, Cl_, time_)

  // Rate Rules:
  ECl_' = k5 * Cl_ * E_ + k9 * ECl - ( k6 * ECl_ + k10 * ECl_ );
end

model renal_anion_exchanger_model__EHCO3_(EHCO3_, k11, k12, k7, k8, E_, EHCO3, HCO3_, time_)

  // Rate Rules:
  EHCO3_' = k7 * HCO3_ * E_ + k11 * EHCO3 - ( k8 * EHCO3_ + k12 * EHCO3_ );
end

model renal_anion_exchanger_model__J_Cl_influx(J_Cl_influx, k9, k10, ECl_, ECl, Cl_, HCO3_, time_)

  // Assignment Rules:
  J_Cl_influx := ( k9 * ECl - k10 * ECl_ ) * power( 1.0 + Cl_ / Ki_Cl + HCO3_ / Ki_HCO3 , -1.0 );

  // Variable initializations:
  Ki_Cl = 0.528;
  Ki_HCO3 = 0.423;
end

model renal_anion_exchanger_model__reaction_constants(k1, k2, k3, k4, k5, k6, k7, k8, k9, k10, k11, k12, Cl, HCO3, Cl_, HCO3_)

  // Variable initializations:
  k1 = 1.0E8;
  k2 = 7.87E6;
  k3 = 1.0E8;
  k4 = 8.28E6;
  k5 = 1.0E8;
  k6 = 7.87E6;
  k7 = 1.0E8;
  k8 = 8.28E6;
  k9 = 5.14E5;
  k10 = 9.26E4;
  k11 = 3.24E5;
  k12 = 5.83E4;
  Cl = 1.0;
  HCO3 = 1.0;
  Cl_ = 1.0;
  HCO3_ = 1.0;
end

model renal_anion_exchanger_model____main()

  // Sub-modules, and any changes to those submodules:
  environment: renal_anion_exchanger_model__environment(time_);
  E: renal_anion_exchanger_model__E(E0, k1, k2, k3, k4, ECl0, EHCO30, HCO3, Cl, time_);
  ECl: renal_anion_exchanger_model__ECl(ECl0, k1, k2, k9, k10, ECl_0, E0, Cl, time_);
  EHCO3: renal_anion_exchanger_model__EHCO3(EHCO30, k11, k12, k3, k4, E0, EHCO3_0, HCO3, time_);
  E_: renal_anion_exchanger_model__E_(E_0, k5, k6, k7, k8, ECl_0, EHCO3_0, HCO3_, Cl_, time_);
  ECl_: renal_anion_exchanger_model__ECl_(ECl_0, k5, k6, k9, k10, ECl0, E_0, Cl_, time_);
  EHCO3_: renal_anion_exchanger_model__EHCO3_(EHCO3_0, k11, k12, k7, k8, E_0, EHCO30, HCO3_, time_);
  J_Cl_influx: renal_anion_exchanger_model__J_Cl_influx(J_Cl_influx0, k9, k10, ECl_0, ECl0, Cl_, HCO3_, time_);
  reaction_constants: renal_anion_exchanger_model__reaction_constants(k1, k2, k3, k4, k5, k6, k7, k8, k9, k10, k11, k12, Cl, HCO3, Cl_, HCO3_);
end

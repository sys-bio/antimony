model renal_H_ATPase_model__EH(EH, k1, k2, k3, k4, E, EH2, Ha, time_)

  // Rate Rules:
  EH' = k1 * Ha * E + k4 * EH2 - ( k2 * EH + k3 * Ha * EH );
end

model renal_H_ATPase_model__E(E, k1, k2, k9, k10, EH, E_, Ha, ATP_, ADP_, Pi_, time_)

  // Rate Rules:
  E' = k10 * ADP_ * Pi_ * E_ + k2 * EH - ( k1 * Ha * E + k9 * ATP_ * E );
end

model renal_H_ATPase_model__EH2(EH2, k11, k12, k3, k4, EH, EH2_, Ha, time_)

  // Rate Rules:
  EH2' = k3 * Ha * EH + k12 * EH2_ - ( k4 * EH2 + k11 * EH2 );
end

model renal_H_ATPase_model__EH_(EH_, k5, k6, k7, k8, E_, EH2_, H_, time_)

  // Rate Rules:
  EH_' = k5 * H_ * E_ + k8 * EH2_ - ( k6 * EH_ + k7 * H_ * EH_ );
end

model renal_H_ATPase_model__E_(E_, k5, k6, k9, k10, EH_, E, H_, ATP_, ADP_, Pi_, time_)

  // Rate Rules:
  E_' = k9 * ATP_ * E + k6 * EH_ - ( k5 * H_ * E_ + k10 * ADP_ * Pi_ * E_ );
end

model renal_H_ATPase_model__EH2_(EH2_, k11, k12, k7, k8, EH_, EH2, H_, time_)

  // Rate Rules:
  EH2_' = k7 * H_ * EH_ + k11 * EH2 - ( k8 * EH2_ + k12 * EH2_ );
end

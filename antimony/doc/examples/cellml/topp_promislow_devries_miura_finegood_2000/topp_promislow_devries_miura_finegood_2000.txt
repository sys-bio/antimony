model topp_2000__environment(time_)
end

model topp_2000__glucose_dynamics(G, time_, I)

  // Rate Rules:
  G' = R0 - (EG0 + SI * I) * G;

  // Variable initializations:
  G = 600;
  R0 = 846;
  SI = 0.72;
  EG0 = 1.44;
end

model topp_2000__insulin_dynamics(I, time_, G, beta)

  // Rate Rules:
  I' = beta / 1 * sigma * power(G, 2 ) / (alpha + power(G, 2 )) - k * I;

  // Variable initializations:
  I = 0;
  alpha = 2000;
  sigma = 43.2;
  k = 432;
end

model topp_2000__beta_cell_dynamics(beta, time_, G)

  // Rate Rules:
  beta' = (r1 * G + -(d0) + -(r2) * power(G, 2 )) * beta;

  // Variable initializations:
  beta = 0;
  d0 = 0.06;
  r1 = 0.84e-3;
  r2 = 0.24e-5;
end

model topp_2000____main()

  // Sub-modules, and any changes to those submodules:
  environment: topp_2000__environment(time_);
  glucose_dynamics: topp_2000__glucose_dynamics(G, time_, I);
  insulin_dynamics: topp_2000__insulin_dynamics(I, time_, G, beta);
  beta_cell_dynamics: topp_2000__beta_cell_dynamics(beta, time_, G);
end

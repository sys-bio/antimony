model wodarz_hamer_model_2007_version01__T(T, delta, gamma, lambda_, v, time_)

  // Rate Rules:
  T' = lambda_ - ( delta * T + gamma * T * v );

  // Variable initializations:
  T = 1000.0;
end

model wodarz_hamer_model_2007_version01__I(I, alpha, gamma, T, v, p, z, w, time_)

  // Rate Rules:
  I' = gamma * T * v - ( alpha * I + p * I * ( z + w ));

  // Variable initializations:
  I = 0.0001;
end

model wodarz_hamer_model_2007_version01__x(x, r, v, y, k, d, beta, time_)

  // Rate Rules:
  x' = r * x * v * ( 1.0 - ( x + y ) / k ) - ( d * x + beta * x * v );

  // Variable initializations:
  x = 10.0;
end

model wodarz_hamer_model_2007_version01__y(y, r, v, x, k, a, p, z, w, beta, time_)

  // Rate Rules:
  y' = beta * x * v + r * y * v * ( 1.0 - ( x + y ) / k ) - ( a * y + p * y * ( z + w ));

  // Variable initializations:
  y = 0.0;
end

model wodarz_hamer_model_2007_version01__v(v, u, eta, y, I, time_)

  // Rate Rules:
  v' = eta * ( y + I ) - u * v;

  // Variable initializations:
  v = 0.0001;
end

model wodarz_hamer_model_2007_version01__y(y, v, time_)

  // Rate Rules:
  y' = r * y * v * ( 1.0 - y / k ) - a * y;

  // Variable initializations:
  y = 0.0;
  r = 1.0;
  a = 0.2;
  k = 10.0;
end

model wodarz_hamer_model_2007_version01__v(v, y, time_)

  // Rate Rules:
  v' = eta * y - u * v;

  // Variable initializations:
  v = 1.0;
  u = 0.5;
  eta = 1.0;
end

model wodarz_hamer_model_2007_version01____main()

  // Sub-modules, and any changes to those submodules:
  environment: wodarz_hamer_model_2007_version01__environment(time_);
  y: wodarz_hamer_model_2007_version01__y(y0, v0, time_);
  v: wodarz_hamer_model_2007_version01__v(v0, y0, time_);
end

model colegrove_albrecht_friel_2000_version01__Ca_extrusion_across_the_plasma_membrane(J_pm, Ca_i, time_)

  // Assignment Rules:
  J_pm := J_leak + J_extru;
  J_leak := k_leak * ( Ca_i - Ca_o );
  J_extru := Vmax_extru / ( 1.0 + power( EC50_extru / Ca_i , n_extru ));

  // Variable initializations:
  k_leak = 0.00000037;
  Vmax_extru = 28.3;
  EC50_extru = 378.8;
  n_extru = 1.8;
  Ca_o = 0.002;
end

model colegrove_albrecht_friel_2000_version01__mitochondrial_Ca_uptake(J_uni, Ca_i, time_)

  // Assignment Rules:
  J_uni := kuni_max * Ca_i / ( 1.0 + power( EC50_uni / Ca_i , n_uni ));

  // Variable initializations:
  kuni_max = 75.9;
  EC50_uni = 10.0;
  n_uni = 2.0;
end

model colegrove_albrecht_friel_2000_version01__mitochondrial_Ca_release(J_NaCa, Ca_inhibited_J_NaCa, J_mito, J_uni, Ca_i, Ca_m, time_)

  // Assignment Rules:
  J_NaCa := -( Vmax_NaCa ) / ( 1.0 + EC50_NaCa / Ca_m );
  Ca_inhibited_J_NaCa := -(( delta_Ca_i * Vmax_NaCa )) / ( 1.0 + EC50_NaCa / Ca_i );
  J_mito := J_uni + J_NaCa;
  delta_Ca_i := 1.0 - 1.0 / ( 1.0 - power( k_inhib / Ca_i , n_inhib ));

  // Variable initializations:
  Vmax_NaCa = -35.0;
  EC50_NaCa = 307.0;
  n_inhib = 6.0;
  k_inhib = 500.0;
end

model colegrove_albrecht_friel_2000_version01__intracellular_calcium(Ca_i, J_pm, J_mito, time_)

  // Rate Rules:
  Ca_i' = -(( J_pm + J_mito ));
end

model colegrove_albrecht_friel_2000_version01__mitochondrial_calcium(Ca_m, J_mito, time_)

  // Assignment Rules:
  gamma := v_m * k_m / ( v_i * k_i );

  // Rate Rules:
  Ca_m' = J_mito / gamma;
end

model colegrove_albrecht_friel_2000_version01____main()

  // Sub-modules, and any changes to those submodules:
  environment: colegrove_albrecht_friel_2000_version01__environment(time_);
  Ca_extrusion_across_the_plasma_membrane: colegrove_albrecht_friel_2000_version01__Ca_extrusion_across_the_plasma_membrane(J_pm, Ca_i, time_);
  mitochondrial_Ca_uptake: colegrove_albrecht_friel_2000_version01__mitochondrial_Ca_uptake(J_uni, Ca_i, time_);
  mitochondrial_Ca_release: colegrove_albrecht_friel_2000_version01__mitochondrial_Ca_release(J_NaCa, Ca_inhibited_J_NaCa, J_mito, J_uni, Ca_i, Ca_m, time_);
  intracellular_calcium: colegrove_albrecht_friel_2000_version01__intracellular_calcium(Ca_i, J_pm, J_mito, time_);
  mitochondrial_calcium: colegrove_albrecht_friel_2000_version01__mitochondrial_calcium(Ca_m, J_mito, time_);
end

  scatchard := (x3 + x4 + x7 + x8 + x9) / x1;

  // Rate Rules:
  x1' = k1_ * x3 - k1 * x1 * x2 + k2_ * x4 - k2 * x1 * x3 + k4_ * (x6 + x7) - k4 * x1 * (x5 + x9) + k5_ * (x7 + x8 + x9) - k5 * x1 * (x5 + x6 + x7);

  // Variable initializations:
  x1 = 10;
end

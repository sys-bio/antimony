model starbuck_1990__Rs(Rs, kfc, kr, ks, kt, krec_c, krec_i, numcell, fr, Ric, Rii, Lo, Cs, time_)

  // Rate Rules:
  Rs' = kr * Cs + krec_c * Ric + krec_i * ( 1.0 - fr ) * Rii + ks - ( kfc / numcell * Rs * Lo + kt * Rs );
end

model starbuck_1990__Cs(Cs, kfc, kr, kc, ku, kt, krec_c, krec_i, numcell, fr, Ts, Ps, Lo, Rs, Cic, Cii, time_)

  // Rate Rules:
  Cs' = ku * Ts + krec_c * Cic + krec_i * ( 1.0 - fr ) * Cii - ( kfc / numcell * Rs * Lo + kr * Cs + kc * Cs * Ps + kt * Cs );
end

model starbuck_1990__Ts(Ts, kc, ku, Cs, Ps, lamda, time_)

  // Rate Rules:
  Ts' = kc * Cs * Ps - ( ku * Ts + lamda * Ts );
end

model starbuck_1990__Ps(Ps, kc, ku, krec_p, Ts, Cs, Pi_, lamda, time_)

  // Rate Rules:
  Ps' = ku * Ts + krec_p * Pi_ - ( kc * Cs * Ps + lamda * Ps );
end

model starbuck_1990__Lo(Lo, kr, kfc, krec_c, Rs, Cs, Lic, numcell, time_)

  // Rate Rules:
  Lo' = numcell * kr * Cs + krec_c * Lic - kfc * Rs * Lo;
end

model starbuck_1990__Rii(Rii, kfe, kr_, kdeg, krec_i, numcell, fr, Lii, Cii, time_)

  // Rate Rules:
  Rii' = kr_ * Cii - ( kfe / numcell * Rii * Lii + krec_i * ( 1.0 - fr ) * Rii + fr * kdeg * Rii );
end

model starbuck_1990__Cii(Cii, kfe, kr_, ku_, kdeg, krec_i, numcell, fr, Ti, Lii, Rii, time_)

  // Rate Rules:
  Cii' = kfe / numcell * Rii * Lii + ku_ * Ti - ( kr_ * Cii + fr * kdeg * Cii + krec_i * ( 1.0 - fr ) * Cii );
end

model starbuck_1990__Ti(Ti, kdeg, ku_, Ts, lamda, time_)

  // Rate Rules:
  Ti' = lamda * Ts - ( ku_ * Ti + kdeg * Ti );
end

model starbuck_1990__Pi(Pi_, ku_, krec_p, Ti, Ps, lamda, time_)

  // Rate Rules:
  Pi_' = ku_ * Ti + lamda * Pi_ - krec_p * Pi_;
end

model starbuck_1990__Lii(Lii, kr_, kfe, kdeg, Rii, Cii, numcell, time_)

  // Rate Rules:
  Lii' = numcell * kr_ * Cii - ( kdeg * Lii + kfe * Rii * Lii );
end

model starbuck_1990__Ric(Ric, kfv, kr, kt, krec_c, numcell, Rs, Lic, Cic, time_)

  // Rate Rules:
  Ric' = kr * Cic + kt * Rs - ( kfv / numcell * Ric * Lic + krec_c * Ric );
end

model starbuck_1990__Cic(Cic, kfv, kr, kt, krec_c, numcell, Cs, Lic, Ric, time_)

  // Rate Rules:
  Cic' = kfv / numcell * Ric * Lic + kt * Cs - ( kr * Cic + krec_c * Cic );
end

model starbuck_1990__Lic(Lic, kr, kfv, krec_c, Ric, Cic, numcell, time_)

  // Rate Rules:
  Lic' = numcell * kr * Cic - ( krec_c * Lic + kfv * Ric * Lic );
end

  ku := kc / Kcp;

model Dupont_1992a__parameters(kf, k, Y, Z, v2, v3)

  // Assignment Rules:
  v2 := VM2 * power(Z, n) / (power(KP, n) + power(Z, n));
  v3 := VM3 * (power(Y, m) / (power(KR, m) + power(Y, m))) * (power(Z, p) / (power(KA, p) + power(Z, p)));

  // Variable initializations:
  VM2 = 65;
  VM3 = 500;
  KR = 2;
  KA = 0.9;
  KP = 1;
  n = 2;
  m = 2;
  p = 4;
  kf = 1;
  k = 10;
end

model Dupont_1992a__cytosol(time_, v2, v3, kf, k, Y, Z)

  // Rate Rules:
  Z' = v0 + v1beta - v2 + v3 + kf * Y - k * Z;

  // Variable initializations:
  v0 = 1;
  v1beta = 2.7;
  Z = 0.26;
end

model Dupont_1992a__insensitive_pool(time_, v2, v3, kf, Y)

  // Rate Rules:
  Y' = v2 - v3 - kf * Y;

  // Variable initializations:
  Y = 1.7;
end

model Dupont_1992a__kinase_reaction(Z, vK)

  // Assignment Rules:
  vK := vMK * (power(Z, q) / (power(Ka, q) + power(Z, q)));

  // Variable initializations:
  vMK = 20;
  Ka = 2.5;
  q = 1;
end

model Dupont_1992a__phosphorylation(time_, vK, Z)

  // Sub-modules, and any changes to those submodules:
  kinase_reaction: Dupont_1992a__kinase_reaction(Z, vK);

  // Rate Rules:
  Wstar' = vP / WT * (vK / vP * (1 - Wstar) / (K1 + 1 - Wstar) - Wstar / (K2 + Wstar));

  // Variable initializations:
  vP = 2.5;
  K1 = 0.01;
  K2 = 0.01;
  WT = 1;
  Wstar = 0.37;
end

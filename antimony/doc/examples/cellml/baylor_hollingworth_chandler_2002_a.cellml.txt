model cellmlmod_environment()

  //Other declarations:
  var time;
end

model cellmlmod_Ca()

  // Rate Rules:
   Ca ' = k1_  *  CaATP  -  k1  *  Ca  *  ATP

  // Variable initializations:
  Ca = 0.05;

  //Other declarations:
  var Ca, k1, k1_, ATP, CaATP, time;
end

model cellmlmod_ATP()

  // Rate Rules:
   ATP ' = k1_  *  CaATP  -  k1  *  Ca  *  ATP

  // Variable initializations:
  ATP = 8000.0;

  //Other declarations:
  var ATP, k1, k1_, Ca, CaATP, time;
end

model cellmlmod_CaATP()

  // Rate Rules:
   CaATP ' = k1  *  Ca  *  ATP  -  k1_  *  CaATP

  // Variable initializations:
  CaATP = 1.0;

  //Other declarations:
  var CaATP, k1, k1_, Ca, ATP, time;
end

model cellmlmod_reaction_constants()

  // Variable initializations:
  k1 = 1.566E7;
  k1_ = 34461.0;

  //Other declarations:
  var k1, k1_;
end

model baylor_model_f_2002()

  // Sub-modules, and any changes to those submodules:
  environment: cellmlmod_environment()
  Ca: cellmlmod_Ca()
  ATP: cellmlmod_ATP()
  CaATP: cellmlmod_CaATP()
  reaction_constants: cellmlmod_reaction_constants()
  Ca.time is environment.time;
  ATP.time is environment.time;
  CaATP.time is environment.time;
  Ca.Ca is ATP.Ca;
  Ca.ATP is ATP.ATP;
  ATP.Ca is CaATP.Ca;
  Ca.CaATP is CaATP.CaATP;
  ATP.ATP is CaATP.ATP;
  ATP.CaATP is CaATP.CaATP;
  Ca.k1 is reaction_constants.k1;
  Ca.k1_ is reaction_constants.k1_;
  ATP.k1 is reaction_constants.k1;
  ATP.k1_ is reaction_constants.k1_;
  CaATP.k1 is reaction_constants.k1;
  CaATP.k1_ is reaction_constants.k1_;
end

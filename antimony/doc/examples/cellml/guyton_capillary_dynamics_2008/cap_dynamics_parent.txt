model capillary_dynamics_model__capillary_pressure(RVS, BFN, PVS, PC)

  // Assignment Rules:
  PC := RVS * 1.7 * BFN + PVS;
end

model capillary_dynamics_model__rate_of_fluid_out_of_capillaries(PC, PGH, PTC, PPC, VTCPL, VTC, CFC)

  // Assignment Rules:
  VTC := (PC - PPC - PGH + PTC) * CFC + VTCPL;
end

end

model capillary_dynamics_model__plasma_volume(DFP, VTC, VTL, VP_init, time_, VP, TRPL)

  // Assignment Rules:
  VPD := VTL - VTC - DFP + TRPL;

  // Rate Rules:
  VP' = VPD;

  // Variable initializations:
  VP = VP_init;
end

model capillary_dynamics_model__plasma_protein_concentration(PRP, VP, CPP)

  // Assignment Rules:
  CPP := PRP / VP;
end

model capillary_dynamics_model__protein_destruction_and_formation(CPP, DLP, CPR, LPPR, LPDE, LPK)

  // Assignment Rules:
  DLP := LPPR - power(CPPD, LPDE) * LPK;
  CPPD := piecewise( 0 , CPP - CPR < 0 , CPP - CPR );
end

model capillary_dynamics_model__plasma_leakage(PC, VTCPL, PCR, CPK, PCE)

  // Assignment Rules:
  VTCPL := power(PRCD * CPK, PCE);
  PRCD := piecewise( 0 , PC - PCR < 0 , PC - PCR );
end

model capillary_dynamics_model__protein_influx_into_interstitium(VTCPL, CPP, CPI, DPC)

  // Assignment Rules:
  DPC := VTCPL * CPP + (CPP - CPI) * 0.00104;
end

model capillary_dynamics_model__total_plasma_protein(DPC, PPD, DPL, DLP, PRP_init, time_, PRP, TRPL)

  // Assignment Rules:
  DPP := DLP + DPL - DPC - PPD + TRPL * 72;

  // Rate Rules:
  PRP' = DPP;

  // Variable initializations:
  PRP = PRP_init;
end

model capillary_dynamics_model__plasma_colloid_osmotic_pressure(CPP, PPC)

  // Assignment Rules:
  PPC := 0.28 * CPP + 0.0019 * power(CPP, 2 );
end

model capillary_dynamics_model__systemic_tissue_fluid_volume_and_protein()
end

model capillary_dynamics_model__total_systemic_fluid_volume(VEC, VP, VPF, VTS)

  // Assignment Rules:
  VTS := VEC - VP - VPF;
end

model capillary_dynamics_model__interstitial_fluid_volume(VTS, VTS2_init, time_, VTS1, TSSLML, TSSLTC)

  // Assignment Rules:
  VTS1 := VTS - VTS2;

  // Rate Rules:
  VTS2' = ((VTS - 12 ) * TSSLML - VTS2) * TSSLTC;

  // Variable initializations:
  VTS2 = VTS2_init;
end

model capillary_dynamics_model__total_interstitial_protein(DPC, DPL, TSP_init, time_, TSP)

  // Assignment Rules:
  DPI := DPC - DPL;

  // Rate Rules:
  TSP' = DPI;

  // Variable initializations:
  TSP = TSP_init;
end

model capillary_dynamics_model__interstitial_protein_concentration(TSP, VTS, CPI)

  // Assignment Rules:
  CPI := TSP / VTS;
end

model capillary_dynamics_model__interstitial_colloid_osmotic_pressure(CPI, PTCPR)

  // Assignment Rules:
  PTCPR := 0.28 * CPI + 0.0019 * power(CPI, 2 );
end

model capillary_dynamics_model__lymph_protein_flow(CPI, VTL, DPL)

  // Assignment Rules:
  DPL := CPI * VTL;
end

model capillary_dynamics_model__hydrostatic_pressure_of_tissue_gel(VTS, PTT, CHY, PGH, HYL, CMPTSS, PGHF)

  // Assignment Rules:
  CHY := power(HYL / VTS / 5 , CMPTSS);
  PGH := CHY * PGHF + PTT;
end

model capillary_dynamics_model__total_osmotic_pressure_of_tissue_gel(CHY, PTCPR, POSHYL, PTC, GCOPF)

  // Assignment Rules:
  POSHYL := CHY * 2;
  PTC := POSHYL * PTCPR * GCOPF;
end

model capillary_dynamics_model__total_tissue_pressure(VTS1, PTT, VTSF)

  // Assignment Rules:
  PTT := power((VTS1 - VTSF) / VTSF, 2 ) * 1;
end

model capillary_dynamics_model__interstial_free_fluid_pressure(PGH, POSHYL, PIF)

  // Assignment Rules:
  PIF := PGH - POSHYL;
end

model capillary_dynamics_model__interstitial_solid_tissue_pressure(PIF, PTT, PTS)

  // Assignment Rules:
  PTS := PTT - PIF;
end

model capillary_dynamics_model__lymph_flow(PIF, PTT, VTL, PLDF)

  // Assignment Rules:
  VTL := piecewise( 0 , PLD < 0 , PLD * 0.02 );
  PLD1 := PIF + PLDF - PTT;
  PLD := piecewise( 7 , PLD1 > 7 , PLD1 );
end

end

model westermark_2003__v_GK(v_GK, Glc)

  // Assignment Rules:
  v_GK := V_GK * power( Glc / S0_5_GK , h_GK ) / ( 1.0 + power( Glc / S0_5_GK , h_GK ));

  // Variable initializations:
  V_GK = 10.0;
  S0_5_GK = 8.0;
  h_GK = 1.7;
end

model westermark_2003__v_GPI(v_GPI, G6P, F6P)

  // Assignment Rules:
  v_GPI := Keq_GPI * G6P * F6P;

  // Variable initializations:
  Keq_GPI = 0.3;
end

model westermark_2003__v_PFK(v_PFK, pi_pfk, FBP, F6P, Glc)

  // Assignment Rules:
  v_PFK := V_PFK * power( Glc / S0_5_PFK , h_PFK ) / ( 1.0 + power( Glc / S0_5_PFK , h_PFK ));
  pi_pfk := FBP / ( S0_5_FBA * S0_5_PFK );
  S0_5_PFK := (power( 0.25 + 2.0 * V_PFK / k_cat_K_PFK , 0.5 ) + 0.5 ) / (power( KT , -1.0 ) + power( KO , -1.0 ) * (power( 0.25 + 2.0 * V_PFK / k_cat_K_PFK , 0.5 ) - 0.5 ));
  sigma := F6P / S0_5_PFK;
  alpha := 5.0 * h_PFK_;
  q := S0_5_FBA / X0_5;
  h_PFK_ := h_PFK - ( h_PFK - h_act ) * ( q * pi_pfk / ( 1.0 + q * pi_pfk ));
  k_cat_K_PFK := V_PFK / 3.0;

  // Variable initializations:
  V_PFK = 100.0;
  S0_5_FBA = 5.0;
  X0_5 = 10.0;
  h_PFK = 2.5;
  h_act = 1.0;
  KT = 2.0;
  KO = 30.0;
end

model westermark_2003__v_FBA(v_FBA, rho, pi_pfk, Keq_TPI)

  // Assignment Rules:
  v_FBA := V_FBA * pi_pfk * ( 1.0 - p * r * power( rho , 2.0 ) / ( pi_pfk * Keq_FBA )) / ( 1.0 + pi_pfk + r * rho + p * r * power( rho , 2.0 ));
  r := S0_5_GAPD / ( Q0_5_FBA * Keq_TPI );
  p := S0_5_GAPD / P0_5_FBA;

  // Variable initializations:
  V_FBA = 25.0;
  Q0_5_FBA = 0.275;
  P0_5_FBA = 0.5;
  S0_5_GAPD = 5.0;
  Keq_FBA = 0.1;
end

model westermark_2003__v_TPI(v_TPI, Keq_TPI, DHAP, G3P)

  // Assignment Rules:
  v_TPI := Keq_TPI * DHAP * G3P;

  // Variable initializations:
  Keq_TPI = 0.045;
end

model westermark_2003__v_GAPD(v_GAPD, rho)

  // Assignment Rules:
  v_GAPD := V_GAPD * rho / ( 1.0 + rho );

  // Variable initializations:
  V_GAPD = 250.0;
end

model wodarz_2007c__x(x, d, gamma, lambda_, beta, v, time_)

  // Rate Rules:
  x' = lambda_ - (d * x + beta * x * v + gamma * x * v);

  // Variable initializations:
  x = 1;
end

model wodarz_2007c__y0(y0, beta, x, a0, eta, phi, L, v, pa, za, p_i, zi, time_)

  // Rate Rules:
  y0' = beta * x * v - (a0 * y0 + eta * y0 + pa * y0 * za + p_i * y0 * zi) + phi * L;

  // Variable initializations:
  y0 = 0;
end

model wodarz_2007c__y1(y1, a1, eta, y0, pa, za, p_i, zi, time_)

  // Variable initializations:

model wodarz_2007c__L(L, gamma, x, v, phi, d, time_)

  // Rate Rules:
  L' = gamma * x * v - (phi * L + d * L);

  // Variable initializations:
  L = 0;
end

model wodarz_2007c__v(v, k, y1, u, time_)

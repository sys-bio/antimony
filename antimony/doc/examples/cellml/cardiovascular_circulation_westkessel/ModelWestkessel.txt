model TempR__TempR(Pi_, Po, Qi, Qo, R)

  // Assignment Rules:
  Pi_ := Po + R * Qi;
  Qo := Qi;
end

model TempRC__TempRC(Pi_, Po, Qi, Qo, R, C, t, P0)

  // Assignment Rules:
  Qo := (Pi_ - Po) / R;

  // Rate Rules:
  Pi_' = (Qi - Qo) / C;

  // Variable initializations:
  Pi_ = P0;
end

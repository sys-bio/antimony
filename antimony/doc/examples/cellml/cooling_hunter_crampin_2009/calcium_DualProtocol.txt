model calcium__calcium(t, Ca)

  // Assignment Rules:
  Ca := CaBeat + CaSigTrans;
  ms_in_pulse := 1000 * (t - floor(t / period) * period);
  baseline := piecewise( 93 , period >= 0.12 , 1121 + (period * 1000 - 16 ) * -(9.8846 ) );
  stimGradient := (1121 - baseline) / 16;
  CaBeat := piecewise( ms_in_pulse * stimGradient + baseline , ms_in_pulse <= 16 , (ms_in_pulse - 16 ) * -(9.8846 ) + 1121 , ( ms_in_pulse > 16) && (ms_in_pulse <= 120 ), baseline );

  // Variable initializations:
  period = 1.02;
  CaSigTrans = 0;
end

model calcium____main()

  // Sub-modules, and any changes to those submodules:
  calcium: calcium__calcium(t, Ca);
end

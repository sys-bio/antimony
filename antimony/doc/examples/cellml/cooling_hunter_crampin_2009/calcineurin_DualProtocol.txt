model calcineurin__calcineurin(t, Ca, Ntot, act_N)

  // Assignment Rules:
  act_N := power(Ca, n) / (power(Ca, n) + power(K_mN, n) * (1 + K_dN / M));

  // Variable initializations:
  M = 6000;
  Ntot = 1000;
  K_mN = 535;
  n = 2.92;
  K_dN = 1760;
end

model calcineurin____main()

  // Sub-modules, and any changes to those submodules:
  calcineurin: calcineurin__calcineurin(t, Ca, Ntot, act_N);
end

model saucerman_model_sarcolemma_2008__Reaction_02(Reaction_02, k02, k20, K, Mg, Ca, CaM, Ca2CaM)

  // Assignment Rules:
  Reaction_02 := k02 * power( Ca , 2 ) * CaM - k20 * Ca2CaM;
  k02 := k20 / Kd02;
  Kd02 := piecewise( 0.0025 * ( 1.0 + K / 0.94 - Mg / 0.012 ) * ( 1.0 + K / 8.1 + Mg / 0.022 ) , Mg <= 1.0 , 0.0025 * ( 1.0 + K / 0.94 + ( Mg - 1.0 ) / 0.06 - 1.0 / 0.012 ) * ( 1.0 + K / 8.1 + 1.0 / 0.022 + ( Mg - 1.0 ) / 0.068 ) );

  // Variable initializations:
  k20 = 10.0;
end

model saucerman_model_sarcolemma_2008__Reaction_24(Reaction_24, k24, k42, K, Mg, Ca, Ca2CaM, Ca4CaM)

  // Assignment Rules:
  Reaction_24 := k24 * power( Ca , 2 ) * Ca2CaM - k42 * Ca4CaM;
  k24 := k42 / Kd24;
  Kd24 := piecewise( 0.128 * ( 1.0 + K / 0.64 + Mg / 0.014 ) * ( 1.0 + K / 13.0 - Mg / 0.153 ) , Mg <= 1.0 , 0.128 * ( 1.0 + K / 0.64 + ( Mg - 1.0 ) / 0.005 ) * ( 1.0 / 0.012 ) * ( 1.0 + K / 13.0 + ( Mg - 1.0 ) / 0.15 - 1.0 / 0.153 ) );

  // Variable initializations:
  k42 = 500.0;
end

model saucerman_model_sarcolemma_2008__Reaction_CKbt(Reaction_CKbt, kPP1, KmPP1, PP1tot, CaMKIItot, T, Pt, Pb)

  // Assignment Rules:
  Reaction_CKbt := kbt * Pb - kPP1 * PP1tot * Pt / ( KmPP1 + CaMKIItot * Pt );
  kbt := 0.055 * T + 0.0074 * power( T , 2.0 ) + 0.015 * power( T , 3.0 );
end

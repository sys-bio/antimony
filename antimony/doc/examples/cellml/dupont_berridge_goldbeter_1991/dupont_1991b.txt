model Dupont_1991b__beta_pulse(time_, beta)

  // Assignment Rules:
  beta := piecewise( 0 , time_ < tp , betaf * exp(-0.2 * (time_ - tp)) , time_ >= tp );

  // Variable initializations:
  betaf = 0.96;
  tp = 4;
end

model Dupont_1991b__parameters(time_, v0, v1, v2, v3, beta, kf, k, Z, Y)

  // Sub-modules, and any changes to those submodules:
  beta_pulse: Dupont_1991b__beta_pulse(time_, beta);

  // Assignment Rules:
  v2 := VM2 * power(Z, n) / (power(K2, n) + power(Z, n));
  v3 := VM3 * (power(Y, m) / (power(KR, m) + power(Y, m))) * (power(Z, p) / (power(KA, p) + power(Z, p)));

  // Variable initializations:
  v0 = 1;
  v1 = 7.3;
  VM2 = 65;
  VM3 = 500;
  KR = 2;
  KA = 0.9;
  kf = 1;
  k = 10;
  K2 = 1;
  n = 2;
  m = 2;
  p = 4;
end

model Dupont_1991b__cytosol(time_, v0, v1, v2, v3, kf, k, beta, Z, Y)

  // Rate Rules:
  Z' = v0 + v1 * beta - v2 + v3 + kf * Y - k * Z;

  // Variable initializations:
  Z = 0.1;
end

model Dupont_1991b__insensitive_pool(time_, v2, v3, kf, Y)

  // Rate Rules:
  Y' = v2 - v3 - kf * Y;

  // Variable initializations:

end

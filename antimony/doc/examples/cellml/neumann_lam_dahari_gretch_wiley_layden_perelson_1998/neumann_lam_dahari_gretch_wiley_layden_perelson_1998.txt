model neumann_1998__environment(time_)
end

model neumann_1998__target_cells(T, eta, V, beta, time_)

  // Rate Rules:
  T' = s - (d * T + (1 - eta) * beta * V * T);

  // Variable initializations:
  T = 1000;
  s = 10.0;
  d = 0.02;
end

model neumann_1998__infected_cells(I, beta, eta, V, T, time_)

  // Rate Rules:
  I' = (1 - eta) * beta * V * T - delta * I;

  // Variable initializations:
  I = 0;
  delta = 0.5;
end

model neumann_1998__viral_load(V, I, time_)

  // Rate Rules:
  V' = (1 - epsilon) * p * I - c * V;

  // Variable initializations:
  V = 9.4e6;
  epsilon = 0.99;
  p = 100.0;
  c = 5.0;
end

model neumann_1998__model_parameters(beta, eta)

  // Variable initializations:
  beta = 3e-7;
  eta = 0.8;
end

model neumann_1998____main()

  // Sub-modules, and any changes to those submodules:
  environment: neumann_1998__environment(time_);
  target_cells: neumann_1998__target_cells(T, eta, V, beta, time_);
  infected_cells: neumann_1998__infected_cells(I, beta, eta, V, T, time_);
  viral_load: neumann_1998__viral_load(V, I, time_);
  model_parameters: neumann_1998__model_parameters(beta, eta);
end

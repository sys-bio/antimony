model goldbeter_1995__environment(time_)
end

model goldbeter_1995__M(M, PN, time_)

  // Rate Rules:
  M' = vs * (power( KI , n ) / (power( KI , n ) + power( PN , n ))) - vm * ( M / ( Km + M ));

  // Variable initializations:
  M = 0.6;
  Km = 0.5;
  vs = 0.76;
  vm = 0.65;
  n = 4.0;
  KI = 1.0;
end

model goldbeter_1995__P0(P0, M, P1, K1, V1, K2, V2, time_)

  // Rate Rules:
  P0' = ks * M - V1 * ( P0 / ( K1 + P0 )) + V2 * ( P1 / ( K2 + P1 ));

  // Variable initializations:
  P0 = 0.5;
  ks = 0.38;
end

model goldbeter_1995__P1(P1, P0, P2, K1, V1, K2, V2, K3, V3, K4, V4, time_)

  // Rate Rules:
  P1' = V1 * ( P0 / ( K1 + P0 )) - ( V2 * ( P1 / ( K2 + P1 )) + V3 * ( P1 / ( K3 + P1 ))) + V4 * ( P2 / ( K4 + P2 ));

  // Variable initializations:
  P1 = 0.6;
end

model goldbeter_1995__P2(P2, P1, PN, k1, k2, K3, V3, K4, V4, time_)

  // Rate Rules:
  P2' = V3 * ( P1 / ( K3 + P1 )) - ( V4 * ( P2 / ( K4 + P2 )) + k1 * P2 ) + k2 * PN - vd * ( P2 / ( Kd + P2 ));

  // Variable initializations:
  P2 = 0.6;
  Kd = 0.2;
  vd = 0.95;
end

model goldbeter_1995__PN(PN, P2, k1, k2, time_)

  // Rate Rules:
  PN' = k1 * P2 - k2 * PN;

  // Variable initializations:
  PN = 1.1;
end

model goldbeter_1995__Pt(Pt, P2, P1, P0, PN)

  // Assignment Rules:
  Pt := P0 + P1 + P2 + PN;
end

model goldbeter_1995__parameters(k1, k2, K1, V1, K2, V2, K3, V3, K4, V4)

  // Variable initializations:
  k1 = 1.9;
  k2 = 1.3;
  K1 = 2.0;
  V1 = 3.2;
  K2 = 2.0;
  V2 = 1.58;
  K3 = 2.0;
  V3 = 5.0;
  K4 = 2.0;
  V4 = 2.5;
end

model goldbeter_1995____main()

  // Sub-modules, and any changes to those submodules:
  environment: goldbeter_1995__environment(time_);
  M: goldbeter_1995__M(M0, PN0, time_);
  P0: goldbeter_1995__P0(P00, M0, P10, K1, V1, K2, V2, time_);
  P1: goldbeter_1995__P1(P10, P00, P20, K1, V1, K2, V2, K3, V3, K4, V4, time_);
  P2: goldbeter_1995__P2(P20, P10, PN0, k1, k2, K3, V3, K4, V4, time_);
  PN: goldbeter_1995__PN(PN0, P20, k1, k2, time_);
  Pt: goldbeter_1995__Pt(Pt0, P20, P10, P00, PN0);
  parameters: goldbeter_1995__parameters(k1, k2, K1, V1, K2, V2, K3, V3, K4, V4);
end

  R14: C1 -> ; compartment_ * kd4 * C1;
  R15:  -> C2; compartment_ * v11 * pow(T2, n) / (pow(k110, n) + pow(T2, n) + pow(C3, n));
  R16: C2 -> ; compartment_ * kd5 * C2;
  R17:  -> C3; compartment_ * v10 * pow(T3, n) / (pow(k100, n) + pow(T3, n) + pow(C1, n));

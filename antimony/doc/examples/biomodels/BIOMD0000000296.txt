  J5: A1 => $sink; environment*(dAA1 + D)*A1;
  J6: $source => A2; environment*kA2*C2;
  J7: A2 => $sink; environment*(dAA2 + D)*A2;

  // Species initializations:
  IPTG = 5;
  C1 = 20;
  C2 = 20;

  A2 = 0.1;

  kc2 = 0.4;

  d2 = 0.3;
  K2 = 10;
  K1 = 10;

  dAA2 = 0.11;

  //Other declarations:
  var kA2, d1;
  const environment, kc1, kc2, Cm, D, kA1, d2, K2, K1, dAA1, dAA2;

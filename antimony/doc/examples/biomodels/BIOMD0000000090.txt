  v18: oah -> ; c1 * k18 * oah;

  v11a2: $C2 + oxy -> $C1 + $H2O; c2 * k11 * N2 * oxy / ((a * N2 + oxy) * (1 + pow(hyd / Kh, m)));

  vLEAK: $Ho -> $Hm; 0;
  v12: A3c -> $A2c; c1 * k12 * A3c;

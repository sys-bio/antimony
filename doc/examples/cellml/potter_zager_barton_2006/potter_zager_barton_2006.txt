model potter_model_2006__ATbl(ATbl, Qb, Qp, Ql, Qt, CTbf, CTblf, CTpf, CTbl, CTlf, CTtv, ts, time_)

  // Rate Rules:
  ATbl' = Qb * ( CTbf - CTblf ) + Qp * ( CTpf - CTbl ) + Ql * ( CTlf - CTbl ) + Qt * ( 1.0 - ts ) * ( CTtv - CTblf ) + k2T;

  // Variable initializations:
  ATbl = 0.0;
  k2T = 0.17;
end

model potter_model_2006__ADbl(ADbl, Qbt, Qp, Ql, CDbf, CDblf, CDpf, CDlf, CDbl, time_)

  // Rate Rules:
  ADbl' = Qbt * ( CDbf - CDblf ) + Qp * ( CDpf - CDblf ) + Ql * ( CDlf - CDbl );

  // Variable initializations:
  ADbl = 0.0;
end

model potter_model_2006__CT_A(CT_A, CTblf, CAf, time_)

  // Rate Rules:
  CT_A' = kTA_on * CTblf * CAf - kTA_off * CT_A;

  // Variable initializations:
  CT_A = 0.0;
  kTA_on = 0.18;
  kTA_off = 8131.0;
end

model potter_model_2006__CD_A(CD_A, CDblf, CAf, time_)

  // Rate Rules:

  // Variable initializations:
  CD_A = 0.0;
  kDA_on = 0.15;
  kDA_off = 5407.0;
end

model potter_model_2006__CTblf(CTblf, CTbl, CT_A)

  // Assignment Rules:
  CTblf := CTbl - CT_A;
end

model potter_model_2006__CDblf(CDblf, CDbl, CD_A)

model potter_model_2006__CAf(CAf, CA, CD_A, CT_A)

model potter_model_2006__ATl(ATl, k1T, km5alpha, Vmax1, Ql, CTbl, CTlf, PTl, time_)

  // Rate Rules:
  ATl' = Ql * ( CTbl - CTlf ) - ( Vmax1 * CTlf / ( km5alpha + CTlf ) + k1T * ( ATl / PTl ));

  // Variable initializations:
  ATl = 0.0;
end

model potter_model_2006__ADl(ADl, k1D, km5alpha, Vmax1, Ql, CDbl, CTlf, CDlf, PDl, time_)

model potter_model_2006__ATif(ATif, k1T, Qt, mu, PTif, CTblf, CTif, CTst, CLH, PTst, ts, time_)

  // Assignment Rules:
  CTpf := ( CTp - ( CT_R + 2.0 * CTT + CDT + T )) / ( 1.0 + NTp );
  T := CTTsec + CDTsec + CTT5a + CDT5a + CTTcd + CDTcd + CTTcp + CDTcp;

  // Variable initializations:

  // Assignment Rules:
  CDpf := ( CDp - ( CD_R + 2.0 * CDD + CDT + D )) / ( 1.0 + NDp );
  D := CDDsec + CDTsec + CDD5a + CDT5a + CDDcd + CDTcd + CDDcp + CDTcp;

  // Variable initializations:
  NDp = 1.00;
end

end

model potter_model_2006__ALH(ALH, CTblf, CDblf, time_)

  // Assignment Rules:
  A := -( kLH1 ) * ( alpha_T * CTblf + alpha_D * CDblf ) + kLH2;
  B := kLH3;

  // Rate Rules:
  ALH' = piecewise( A - kLH4 * ALH , A > B , B - kLH4 * ALH );

  // Variable initializations:
  ALH = 0.0;
  kLH1 = 0.13;
  kLH2 = 0.026;
  kLH3 = 1.68E-4;
  kLH4 = 0.80;
  alpha_T = 0.25;
  alpha_D = 0.75;
end

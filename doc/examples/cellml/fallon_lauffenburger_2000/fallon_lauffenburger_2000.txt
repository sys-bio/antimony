model fallon_lauffenburger_2000__Rs(Rs, L, Cs, Vs, kf, kr, kt, ksyn, time_)

  // Rate Rules:
  Rs' = (kr + ksyn) * Cs + Vs - (kf * L * Rs + kt * Rs);

  // Variable initializations:
  Rs = 1500;
end

model fallon_lauffenburger_2000__Cs(Cs, L, Rs, kf, kr, ke, time_)

  // Rate Rules:
  Cs' = kf * L * Rs - (kr + ke) * Cs;

  // Variable initializations:
  Cs = 1;
end

model fallon_lauffenburger_2000__Ri(Ri, Li, Ci, Rs, kfe, kre, kt, kh, time_)

  // Rate Rules:
  Ri' = kre * Ci + kt * Rs - (kfe * Li * Ri + kh * Ri);

  // Variable initializations:
  Ri = 300;
end

model fallon_lauffenburger_2000__Ci(Ci, Li, Ri, Cs, kfe, kre, ke, kh, time_)

  // Rate Rules:
  Ci' = kfe * Li * Ri + ke * Cs - (kre + kh) * Ci;

  // Variable initializations:
  Ci = 1;
end

model fallon_lauffenburger_2000__Li(Li, Ci, Ri, kfe, kre, kx, Ve, NA, time_)

  // Rate Rules:
  Li' = (kre * Ci - kfe * Li * Ri) / (Ve * NA) - kx * Li;

  // Variable initializations:
  Li = 0.01;
end

model fallon_lauffenburger_2000__Ld(Ld, Ci, kh, time_)

  // Rate Rules:
  Ld' = kh * Ci;

  // Variable initializations:
  Ld = 1;
end

model fallon_lauffenburger_2000__L(L, Rs, Cs, Li, Y, kf, kr, kx, Ve, NA, time_)

  // Rate Rules:
  L' = (kr * Cs + kx * Li * Ve * NA - kf * L * Rs) * Y / NA;

  // Variable initializations:
  L = 10;
end

model fallon_lauffenburger_2000__Y(Y, Cs, time_)

  // Rate Rules:
  Y' = piecewise( (600 * Cs / (250 + Cs) - 200 ) * 1E3 , 600 * Cs / (250 + Cs) - 200 > 0 , 0 );

  // Variable initializations:
  Y = 2.5e8;
end

model fallon_lauffenburger_2000__model_parameters(kr, kf, kre, kfe, kt, Vs, ksyn, ke, kx, kh, Ve, NA)

  // Assignment Rules:
  kf := piecewise( kr / 11.1 , IL2 == 1 , kr / 8.2 );
  kre := piecewise( kr * 8 , IL2 == 1 , kr * 5 );
  kfe := piecewise( kre / 1000 , IL2 == 1 , kre / 3000 );

  // Variable initializations:
  kr = 0.0138;
  kt = 0.007;
  Vs = 11;
  ksyn = 0.0011;
  ke = 0.04;
  kx = 0.15;
  kh = 0.035;
  Ve = 1e-14;
  NA = 6.022e11;
  IL2 = 1;
end

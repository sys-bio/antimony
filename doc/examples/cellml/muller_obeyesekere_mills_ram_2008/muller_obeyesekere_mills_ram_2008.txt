model muller_model_2008_version01__y1(y1, time_)

  // Assignment Rules:
  g1 := piecewise( 0.0 , ( time_ >= 0.0) && (time_ < 5.0 ), 1.0 , ( time_ >= 5.0) && (time_ <= 10.0 ), 0.0 );

  // Rate Rules:
  y1' = a1 * ( g1 / ( b1 + g1 )) - d1 * y1;

  // Variable initializations:
  y1 = 0.0;
  a1 = 10.0;
  b1 = 10.0;
  d1 = 0.2;
end

  // Rate Rules:

  f53 = 1.5;
  f13 = 0.6;
